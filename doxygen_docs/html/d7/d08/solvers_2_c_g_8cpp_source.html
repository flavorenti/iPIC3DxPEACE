39bool CG(double *xkrylov, int xkrylovlen, double *b, int maxit, double tol, FIELD_IMAGE FunctionImage, Field * field) {
40 // allocate residual, image, p, b, calculated on central points
41 double *r = new double[xkrylovlen];
42 double *v = new double[xkrylovlen];
43 double *z = new double[xkrylovlen];
44 double *im = new double[xkrylovlen];
45 double c, t, d, initial_error;
46 eqValue(0.0, r, xkrylovlen);
47 eqValue(0.0, v, xkrylovlen);
48 eqValue(0.0, z, xkrylovlen);
49 eqValue(0.0, im, xkrylovlen);
50
51 MPI_Comm fieldcomm = (field->get_vct()).getFieldComm();
52
53 int i = 0;
54 bool CONVERGED = false;
55 bool CGVERBOSE = false;
56 // initial guess for x: all the components are equal to 0
57 eqValue(0, xkrylov, xkrylovlen);
58 // Compute r = b -Ax
59 (field->*FunctionImage) (im, xkrylov);
60 sub(r, b, im, xkrylovlen);
61 // v = r
62 eq(v, r, xkrylovlen);
63 c = dotP(r, r, xkrylovlen,&fieldcomm);
64 initial_error = sqrt(c);
65 if (is_output_thread())
66 printf("CG Initial error: %g\n", initial_error);
67 //cout << "CG Initial error: " << initial_error << endl;
68 if (initial_error < 1E-16)
69 return (true);
70 while (i < maxit) {
71 (field->*FunctionImage) (z, v);
72 t = c / dotP(v, z, xkrylovlen,&fieldcomm);
73 // x(i+1) = x + t*v
74 addscale(t, xkrylov, v, xkrylovlen);
75 // r(i+1) = r - t*z
76 addscale(-t, r, z, xkrylovlen);
77 d = dotP(r, r, xkrylovlen,&fieldcomm);
78 if (CGVERBOSE && is_output_thread())
79 {
80 printf("Iteration # %d - norm of residual relative to initial error %g\n",
81 i, sqrt(d) / initial_error);
82 //cout << "Iteration # " << i << " - norm of residual relative to initial error " << sqrt(d) / initial_error << endl;
83 }
84 if (sqrt(d) < tol * initial_error) {
85 // if (d < tol){
86
87 if (is_output_thread())
88 {
89 printf("CG converged at iteration # %d with error %g\n", i, sqrt(d));
90 //cout << "CG converged at iteration # " << i << " with error " << sqrt(d) << endl;
91 }
92 CONVERGED = true;
93 break;
94 }
95 else if (sqrt(d) > 10E8 * initial_error) {
96 if (is_output_thread()) {
97 printf("CG not converged after %d iterations\n", maxit);
98 //cerr << "CG not converging" << endl;
99 //cerr << "CG stopped" << endl;
100
101 }
102 CONVERGED = false;
103 return (CONVERGED);
104 break;
105
106 }
107
108 // calculate the new v
109 addscale(1, d / c, v, r, xkrylovlen);
110 c = d;
111 i++;
112
113 }
114 if (i == maxit)
115 {
116 printf("CG not converged after %d iterations\n", maxit);
117 //cout << "CG not converged after " << maxit << " iterations" << endl;
118 }
119 // deallocate
120 delete[]r;
121 delete[]im;
122 delete[]v;
123 delete[]z;
124 return (CONVERGED);
125}
