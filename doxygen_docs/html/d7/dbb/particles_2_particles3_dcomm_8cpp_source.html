495inline bool Particles3Dcomm::send_pcl_to_appropriate_buffer(
496 SpeciesParticle& pcl, int count[6])
497{
498 int was_sent = true;
499 // put particle in appropriate communication buffer if exiting
500 if(pcl.get_x() < xstart)
501 {
502 sendXleft.send(pcl);
503 count[0]++;
504 }
505 else if(pcl.get_x() > xend)
506 {
507 sendXrght.send(pcl);
508 count[1]++;
509 }
510 else if(pcl.get_y() < ystart)
511 {
512 sendYleft.send(pcl);
513 count[2]++;
514 }
515 else if(pcl.get_y() > yend)
516 {
517 sendYrght.send(pcl);
518 count[3]++;
519 }
520 else if(pcl.get_z() < zstart)
521 {
522 sendZleft.send(pcl);
523 count[4]++;
524 }
525 else if(pcl.get_z() > zend)
526 {
527 sendZrght.send(pcl);
528 count[5]++;
529 }
530 else was_sent = false;
531
532 return was_sent;
533}

547void Particles3Dcomm::apply_periodic_BC_global(
548 vector_SpeciesParticle& pcl_list, int pstart)
549{
550 const double Lxinv = 1/Lx;
551 const double Lyinv = 1/Ly;
552 const double Lzinv = 1/Lz;
553 // apply shift to all periodic directions
554 for(int pidx=pstart;pidx<pcl_list.size();pidx++)
555 {
556 SpeciesParticle& pcl = pcl_list[pidx];
557 if(vct->getPERIODICX_P())
558 {
559 double& x = pcl.fetch_x();
560 x = modulo(x, Lx, Lxinv);
561 }
562 if(vct->getPERIODICY_P())
563 {
564 double& y = pcl.fetch_y();
565 y = modulo(y, Ly, Lyinv);
566 }
567 if(vct->getPERIODICZ_P())
568 {
569 double& z = pcl.fetch_z();
570 z = modulo(z, Lz, Lzinv);
571 }
572 }
573}

575// routines for sorting list of particles
576//
577// sort_pcls: macro to put all particles that satisfy a condition
578// at the end of an array of given size. It has been written so
579// that it could be replaced with a generic routine that takes
580// the "condition" method as a callback function (and if the
581// optimizer is good, the performance in this case would actually
582// be just as good, so maybe we should just make this change).
583//
584// pcls: SpeciesParticle* or vector_SpeciesParticle
585// size_in: number of elements in pcls list
586// start_in: starting index of list to be sorted
587// start_out: returns starting index of particles
588// for which the condition is true.
589// condition: a (probably inline) function of SpeciesParticle
590// that returns true if the particle should go at the end of the list
591//
592#define sort_pcls(pcls, start_in, start_out, condition) \
593{ \

923// receive, sort, and, as appropriate, resend incoming particles
924//
925// assumes that flush_send() has been called
926//
927// returns number of particles that were resent
928//
929int Particles3Dcomm::handle_received_particles(int pclCommMode)
930{
931 using namespace PclCommMode;
932 // we expect to receive at least one block from every
933 // communicator, so make sure that all receive buffers are
934 // clear and waiting
935 //
936 recvXleft.recv_start(); recvXrght.recv_start();
937 recvYleft.recv_start(); recvYrght.recv_start();
938 recvZleft.recv_start(); recvZrght.recv_start();
939
940 // make sure that current block in each sender is ready for sending
941 //
942 sendXleft.send_start(); sendXrght.send_start();
943 sendYleft.send_start(); sendYrght.send_start();
944 sendZleft.send_start(); sendZrght.send_start();
945
946 const int num_recv_buffers = 6;
947
948 int recv_count[6]={0,0,0,0,0,0};
949 int send_count[6]={0,0,0,0,0,0};
950 int num_pcls_recved = 0;
951 int num_pcls_resent = 0;
952 //const int direction_map[6]={
953 // Connection::XDN,Connection::XUP,
954 // Connection::YDN,Connection::YUP,
955 // Connection::ZDN,Connection::ZUP};
956 // receive incoming particles,
957 // immediately resending any exiting particles
958 //
959 MPI_Request recv_requests[num_recv_buffers] =
960 {
961 recvXleft.get_curr_request(), recvXrght.get_curr_request(),
962 recvYleft.get_curr_request(), recvYrght.get_curr_request(),
963 recvZleft.get_curr_request(), recvZrght.get_curr_request()
964 };
965 BlockCommunicator<SpeciesParticle>* recvBuffArr[num_recv_buffers] =
966 {
967 &recvXleft, &recvXrght,
968 &recvYleft, &recvYrght,
969 &recvZleft, &recvZrght
970 };
971
972 assert(!recvXleft.comm_finished());
973 assert(!recvXrght.comm_finished());
974 assert(!recvYleft.comm_finished());
975 assert(!recvYrght.comm_finished());
976 assert(!recvZleft.comm_finished());
977 assert(!recvZrght.comm_finished());
978
979 // determine the periodicity shift for each incoming buffer
980 const bool apply_shift[num_recv_buffers] =
981 {
982 vct->isPeriodicXlower_P(), vct->isPeriodicXupper_P(),
983 vct->isPeriodicYlower_P(), vct->isPeriodicYupper_P(),
984 vct->isPeriodicZlower_P(), vct->isPeriodicZupper_P()
985 };
986 const bool do_apply_BCs[num_recv_buffers] =
987 {
988 vct->noXleftNeighbor_P(), vct->noXrghtNeighbor_P(),
989 vct->noYleftNeighbor_P(), vct->noYrghtNeighbor_P(),
990 vct->noZleftNeighbor_P(), vct->noZrghtNeighbor_P()
991 };
992 const int direction[num_recv_buffers] =
993 {
994 Direction::XDN, Direction::XUP,
995 Direction::YDN, Direction::YUP,
996 Direction::ZDN, Direction::ZUP
997 };
998 // The documentation in the input file says that boundary conditions
999 // are simply ignored in the periodic case, so I omit this check.
1000 //for(int i=0;i<6;i++)assert(!(apply_shift[i]&&do_apply_BCs[i]));
1001 // while there are still incoming particles
1002 // put them in the appropriate buffer
1003 //
1004 while(!(
1005 recvXleft.comm_finished() && recvXrght.comm_finished() &&
1006 recvYleft.comm_finished() && recvYrght.comm_finished() &&
1007 recvZleft.comm_finished() && recvZrght.comm_finished()))
1008 {
1009 int recv_index;
1010 MPI_Status recv_status;
1011 MPI_Waitany(num_recv_buffers,recv_requests,&recv_index,&recv_status);
1012 if(recv_index==MPI_UNDEFINED)
1013 eprintf("recv_requests contains no active handles");
1014 assert_ge(recv_index,0);
1015 assert_lt(recv_index,num_recv_buffers);
1016 //
1017 // grab the received block of particles and process it
1018 //
1019 BlockCommunicator<SpeciesParticle>* recvBuff = recvBuffArr[recv_index];
1020 Block<SpeciesParticle>& recv_block
1021 = recvBuff->fetch_received_block(recv_status);
1022 vector_SpeciesParticle& pcl_list = recv_block.fetch_block();
1023
1024 if(pclCommMode&do_apply_BCs_globally)
1025 {
1026 apply_BCs_globally(pcl_list);
1027 }
1028 else
1029 {
1030 apply_BCs_locally(pcl_list, direction[recv_index],
1031 apply_shift[recv_index], do_apply_BCs[recv_index]);
1032 }
1033
1034 recv_count[recv_index]+=recv_block.size();
1035 num_pcls_recved += recv_block.size();
1036 // process each particle in the received block.
1037 {
1038 for(int pidx=0;pidx<recv_block.size();pidx++)
1039 {
1040 SpeciesParticle& pcl = recv_block[pidx];
1041 bool was_sent = send_pcl_to_appropriate_buffer(pcl, send_count);
1042
1043 if(__builtin_expect(was_sent,false))
1044 {
1045 num_pcls_resent++;
1046 if(pclCommMode&print_sent_pcls)
1047 {
1048 print_pcl(pcl,ns);
1049 }
1050 }
1051 else
1052 {
1053 // The particle belongs here, so put it in the
1054 // appropriate place. For now, all particles are in a
1055 // single list, so we append to the list.
1056 _pcls.push_back(pcl);
1057 }
1058 }
1059 }
1060 // release the block and update the receive request
1061 recvBuff->release_received_block();
1062 recv_requests[recv_index] = recvBuff->get_curr_request();
1063 }
1064
1065 if(print_pcl_comm_counts)
1066 {
1067 dprintf("spec %d recved_count: %d+%d+%d+%d+%d+%d=%d", ns,
1068 recv_count[0], recv_count[1], recv_count[2],
1069 recv_count[3], recv_count[4], recv_count[5], num_pcls_recved);
1070 dprintf("spec %d resent_count: %d+%d+%d+%d+%d+%d=%d", ns,
1071 send_count[0], send_count[1], send_count[2],
1072 send_count[3], send_count[4], send_count[5], num_pcls_resent);
1073 }
1074 // return the number of particles that were resent
1075
1076 return num_pcls_resent;
1077}

1479// communicate particles and apply boundary conditions
1480// until every particle is in the process of its subdomain.
1481//
1482// At the end of this method, the position of every particle in
1483// this process must lie in this process's proper subdomain,
1484// for two reasons: (1) sumMoments assumes that all particles
1485// lie in the proper subdomain of the process, and if this
1486// assumption is violated memory corruption can result, and
1487// (2) the extrapolation algorithm used by the mover is unstable,
1488// which could cause the velocity of particles not in the proper
1489// subdomain to blow up.
1490//
1491// The algorithm proceeds in three steps:
1492// (1) for min_num_iterations, receive particles from neighbor
1493// processes, apply boundary conditions locally (i.e. only if
1494// this is a boundary process), and resend any particles that
1495// still do not belong in this subdomain,
1496// (2) apply boundary conditions globally (i.e. independent of
1497// whether this is a boundary process), and resend any particles
1498// not in this subdomain, and
1499// (3) communicate particles (as many as XLEN+YLEN+ZLEN times)
1500// until every particle is in its appropriate subdomain.
1501//
1502// This communication algorithm is perhaps more sophisticated
1503// than is justified by the mover, which does not properly
1504// resolve fast-moving particles.
1505//
1506// min_num_iterations: number of iterations that this process
1507// applies boundary conditions locally.
1508// This method then applies boundary conditions globally to each
1509// incoming particle until the particle resides in the domain.

1757long long *Particles3Dcomm::getVelocityDistribution(int nBins, double maxVel) {
1758 long long *f = new long long[nBins];
1759 for (int i = 0; i < nBins; i++)
1760 f[i] = 0;
1761 double Vel = 0.0;
1762 double dv = maxVel / nBins;

1764 for (int i = 0; i < _pcls.size(); i++) {
1765 SpeciesParticle& pcl = _pcls[i];
1766 const double u = pcl.get_u();
1767 const double v = pcl.get_v();
1768 const double w = pcl.get_w();
1769 Vel = sqrt(u*u + v*v + w*w);
1770 bin = int (floor(Vel / dv));
1771 if (bin >= nBins)
1772 f[nBins - 1] += 1;
1773 else
1774 f[bin] += 1;
1775 }
1776 MPI_Allreduce(MPI_IN_PLACE, f, nBins, MPI_LONG_LONG, MPI_SUM, mpi_comm);
