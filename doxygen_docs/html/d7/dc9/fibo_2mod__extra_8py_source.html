20 def make_spect(self,
21 tar_data,
22 raw = False):
23
24 """
25 Calculates fourier transform of some variable (real-valued)
26
27 Parameters :
28 - tar_data [fibo.data] target field
29 - raw = False [bool] if False gives calc_slided(fou_data,nx//2,ny//2,nz//2)
30
31 Returns :
32 - fou_data [new fibo.data]
33
34 """
35
36 #create a raw vector (symmetric part of our variable goes to real,
37 # antisymmetric part goes to imaginary, normalization to get integral ...)
38 nx,ny,nz = self.meta['nnn']
39 fou_data = np.absolute(np.fft.fftn(self.get_data(tar_data))) * 2. / (nx*ny*nz)
40
41 if not raw :
42 if nx != 1 : fou_data = np.tile(fou_data,(2,1,1))
43 if ny != 1 : fou_data = np.tile(fou_data,(1,2,1))
44 if nz != 1 : fou_data = np.tile(fou_data,(1,1,2))
45
46 fou_data = fou_data[nx//2:nx+nx//2,ny//2:ny+ny//2,nz//2:nz+nz//2]
47
48 return fou_data

51 def make_plane(self,
52 normal):
53
54 """
55 Calculates two in-plane vectors given the plane normal
56
57 Parameters :
58 - normal [float,float,float] normal direction
59
60 Returns :
61 - versor_one [float,float,float] first in-plane versor
62 - versor_two [float,float,float] second in-plane versor
63 """
64
65 # first, make the normal unitary
66 normal /= np.sqrt((normal**2).sum())
67
68 # second, check for the direction which is closest to the plane normal
69 # if this is the same as the normal, take one of the two perpendiculars
70 quasi_normal = np.zeros(3)
71 quasi_normal[np.argmax(np.absolute(normal))] = 1.
72 if np.all(quasi_normal == normal) :
73 quasi_normal = np.tile(quasi_normal,(2))[1:4]
74
75 # third, find the two in-plane versors by applying the cross product
76 vec_one = np.cross(quasi_normal,normal)
77 vec_one /= np.sqrt((vec_one**2).sum())
78 vec_two = np.cross(normal,vec_one)
79 vec_two /= np.sqrt((vec_two**2).sum())
80
81 return vec_one, vec_two

143 - vec_L [np.ndarray(3)]

147
148 arrx = np.ndarray.flatten(self.get_data(tar_var_x))

151
152 matr = np.zeros([3,3])
153 matr[0,0] = np.mean(arrx * arrx) - np.square(np.mean(arrx))
154 matr[1,1] = np.mean(arry * arry) - np.square(np.mean(arry))
155 matr[2,2] = np.mean(arrz * arrz) - np.square(np.mean(arrz))
156 matr[0,1] = matr[1,0] = np.mean(arrx * arry) - np.mean(arrx)*np.mean(arry)
157 matr[0,2] = matr[2,0] = np.mean(arrx * arrz) - np.mean(arrx)*np.mean(arrz)
158 matr[1,2] = matr[2,1] = np.mean(arry * arrz) - np.mean(arry)*np.mean(arrz)
159
160 vals, vecs = np.linalg.eigh(matr)
161 vec_N, vec_M, vec_L = vecs[:,0], vecs[:,1], vecs[:,2]
162 vec_N = np.reshape(vec_N, (3,1))
163 vec_M = np.reshape(vec_M, (3,1))
164 vec_L = np.reshape(vec_L, (3,1))
165
166 #post-processing

168 if pos_perm is 'LMN' :
169 vec_N[0],vec_N[1],vec_N[2] = self.calc_cross(vec_L[0],vec_M[0],vec_L[1],vec_M[1],vec_L[2],vec_M[2])
170 if pos_perm is 'NML' :
