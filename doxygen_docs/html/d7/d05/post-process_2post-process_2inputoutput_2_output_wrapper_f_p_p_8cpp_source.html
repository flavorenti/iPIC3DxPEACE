29void OutputWrapperFPP::init_output_files(
30 Collective *col,
31 VCtopology3D *vct,
32 Grid3DCU *grid,
33 EMfields3D *EMf,
34 Particles3D *part,
35 int ns,
36 Particles3D *testpart,
37 int nstestpart)
38{
39#ifndef NO_HDF5
40 cartesian_rank = vct->getCartesian_rank();
41 stringstream num_proc_ss;
42 num_proc_ss << cartesian_rank;
43 string num_proc_str = num_proc_ss.str();
44 SaveDirName = col->getSaveDirName();
45 RestartDirName = col->getRestartDirName();
46 int restart_status = col->getRestart_status();
47 output_file = SaveDirName + "/proc" + num_proc_str + ".hdf";
48 restart_file= SaveDirName + "/restart"+ num_proc_str + ".hdf";
49
50 // Initialize the output (simulation results and restart file)
51 hdf5_agent.set_simulation_pointers(EMf, grid, vct, col);
52
53 for (int i = 0; i < ns; ++i){
54 hdf5_agent.set_simulation_pointers_part(&part[i]);
55 }
56 for (int i = 0; i < nstestpart; ++i){
57 hdf5_agent.set_simulation_pointers_part(&testpart[i]);
58 }
59
60 // Add the HDF5 output agent to the Output Manager's list
61 output_mgr.push_back(&hdf5_agent);
62
63 if(col->getWriteMethod() == "shdf5"||(col->getWriteMethod()=="pvtk"&&!col->particle_output_is_off()) ){
64 if (cartesian_rank == 0 && restart_status < 2) {
65 hdf5_agent.open(SaveDirName + "/settings.hdf");
66 output_mgr.output("collective + total_topology + proc_topology", 0);
67 hdf5_agent.close();
68 }
69
70 if (restart_status == 0) {
71 hdf5_agent.open(output_file);
72 }else {
73 hdf5_agent.open_append(output_file);
74 }
75 output_mgr.output("proc_topology ", 0);
76 hdf5_agent.close();
77 }
78
79 if(col->getCallFinalize() || col->getRestartOutputCycle()>0){
80
81 if (cartesian_rank == 0 && restart_status < 2) {
82 hdf5_agent.open(RestartDirName + "/settings.hdf");
83 output_mgr.output("collective + total_topology + proc_topology", 0);
84 hdf5_agent.close();
85 }
86
87 if (restart_status == 0) {
88 hdf5_agent.open(restart_file);
89 hdf5_agent.close();
90 }
91
92 }
93
94
95#endif
96}

116void OutputWrapperFPP::append_restart(int cycle)
117{
118#ifndef NO_HDF5
119 hdf5_agent.open_append(restart_file);
120 output_mgr.output("proc_topology ", cycle);
121 output_mgr.output("Eall + Ball + rhos + Js + pressure", cycle);
122 output_mgr.output("position + velocity + q + ID", cycle, 0);
123 output_mgr.output("testpartpos + testpartvel + testpartcharge", cycle, 0);
124 output_mgr.output("last_cycle", cycle);
125 hdf5_agent.close();
126#endif
127}
