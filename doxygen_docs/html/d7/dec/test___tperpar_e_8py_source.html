2# Code based on fibo library that (1) READS, (2) COMPUTE Tper,Tpar,Epar,Eper,
3# |Eper+VixB|,|Eper+VexB|, and (3) PRINT the new fields. Uses the derived two-dimensional
4# and three-dimensional output of an iPIC3D simu.
5# The new fields are saved in the same dir as the data, with the same dimensions.
6# Add global path to data dir in command line.
7# Only parameters to change are:
8#
9# inp_tars = [str] target variables you want to use (in order!)
10# out_tars = [str] name of the variables creates to print (only scalars!) must follow the order!
11# cuts = [str] cuts to use if the data are 2D (eg data1), if [''] data are 3D (eg data2)
12# cycle_min = [int] minimum cycle to read
13# cycle_max = [int] maximum cycle to read
14#
15# Extra rules for out_tars: Epar require Eper, Eprime require Eper, Tper require Tiso,
16# Tpar require Tper
17#
18# Job 12/2021
19#-------------------------------------------------------------------
20
21######################################################################################
22#======regulate=parameters=============================================================
23inp_tars = ['B','E','Je','Ji','Ve','Vi','TXX','TXY','TXZ','TYY','TYZ','TZZ']
24out_tars = ['Eper','Epar','Eprimei','Eprimee','Jper','Jpar','JdotE','JdotEe','JdotEi','Tiso','Tper','Tpar']
25cuts = ['eq','dp']
26cycle_min = 0
27cycle_max = 2000000
28#=======================================================================================
29########################################################################################
30
31########################################################################################
32####################### DO NOT TOUCH BELOW! ############################################

34#----import-modules---------------
35import fibo as fb
36import time
37import sys
38import numpy as np
39
40#----path-to-run-------------------
41data_address = sys.argv[1]
42
43#----create-your-objects-----------
44fibo_name = 'ipic3d'
45alldata = fb.fibo(fibo_name)
46from_VTK = fb.from_VTK(data_address)
47
48#----loop-on-types-of-data--------
49for cut in cuts:
50
51 #----load-metadata------------------
52 from_VTK.get_meta(silent=False)
53 if cut == 'eq':
54 from_VTK.meta['nz']=1
55 from_VTK.meta['nnn']=(from_VTK.meta['nx'],from_VTK.meta['ny'],1)
56 if cut == 'dp':
57 from_VTK.meta['ny']=1
58 from_VTK.meta['nnn']=(from_VTK.meta['nx'],1,from_VTK.meta['nz'])
59 alldata.meta = from_VTK.meta
60
61 #----load-everything-in-code-units-------------------
62 for seg in alldata.meta['segcycles']:
63 t0 = time.time()
64 if cycle_min<=seg<=cycle_max :
65 print('LOAD->',seg)
66 for tar in inp_tars:
67 if len(tar)<3 :
68 from_VTK.get_vect(alldata.meta['name']+'_'+tar+cut+'_'+str(seg),seg,fibo_obj=alldata,tar_var=tar,silent=True)
69 else :
70 for sp in alldata.meta['species']:
71 from_VTK.get_scal(alldata.meta['name']+'_'+tar+sp+cut+'_'+str(seg),seg,fibo_obj=alldata,tar_var=tar+sp,silent=True)
72
73 print(alldata.data.keys())
74
75#----compute-derived-fields------------------
76 print('COMPUTE->',seg)
77 for tar in out_tars:
78 if tar=='Eper' :
79 alldata.calc_par_per('E_x','B_x','E_y','B_y','E_z','B_z',seg=seg)
80 mod2Eper = alldata.calc_scalr('EperB_x','EperB_x','EperB_y','EperB_y','EperB_z','EperB_z',seg=seg,fill=False)
81 alldata.data[tar+'%.8i'%int(seg)] = np.sqrt( mod2Eper )
82 if tar=='Epar' :
83 mod2Epar = alldata.calc_scalr('EparB_x','EparB_x','EparB_y','EparB_y','EparB_z','EparB_z',seg=seg,fill=False)
84 alldata.data[tar+'%.8i'%int(seg)] = np.sqrt( mod2Epar )
85 if tar=='Eprimei' :
86 alldata.calc_cross('Vi_x','B_x','Vi_y','B_y','Vi_z','B_z',seg=seg)
87 mod2VixB = alldata.calc_scalr('VixB_x','VixB_x','VixB_y','VixB_y','VixB_z','VixB_z',seg=seg,fill=False)
88 alldata.data[tar+'%.8i'%int(seg)] = np.sqrt( mod2Eper+mod2VixB+2.*alldata.calc_scalr('EperB_x','VixB_x','EperB_y','VixB_y','EperB_z','VixB_z',seg=seg,fill=False) )
89 if tar=='Eprimee' :
90 alldata.calc_cross('Ve_x','B_x','Ve_y','B_y','Ve_z','B_z',seg=seg)
91 mod2VexB = alldata.calc_scalr('VexB_x','VexB_x','VexB_y','VexB_y','VexB_z','VexB_z',seg=seg,fill=False)
92 alldata.data[tar+'%.8i'%int(seg)] = np.sqrt( mod2Eper+mod2VexB+2.*alldata.calc_scalr('EperB_x','VexB_x','EperB_y','VexB_y','EperB_z','VexB_z',seg=seg,fill=False) )
93 if tar=='Jper' or tar=='Jpar':
94 alldata.data['J_x%.8i'%seg]=alldata.data['Je_x%.8i'%seg]+alldata.data['Ji_x%.8i'%seg]
95 alldata.data['J_y%.8i'%seg]=alldata.data['Je_y%.8i'%seg]+alldata.data['Ji_y%.8i'%seg]
96 alldata.data['J_z%.8i'%seg]=alldata.data['Je_z%.8i'%seg]+alldata.data['Ji_z%.8i'%seg]
97 alldata.calc_par_per('J_x','B_x','J_y','B_y','J_z','B_z',seg=seg)
98 if tar=='JdotE' :
99 alldata.data[tar+'%.8i'%int(seg)] = alldata.calc_scalr('J_x','E_x','J_y','E_y','J_z','E_z',seg=seg,fill=False)
100 if tar=='JdotEi' :
101 alldata.data[tar+'%.8i'%int(seg)] = alldata.data['JdotE%.8i'%int(seg)] + alldata.calc_scalr('J_x','VixB_x','J_y','VixB_y','J_z','VixB_z',seg=seg,fill=False)
102 if tar=='JdotEe' :
103 alldata.data[tar+'%.8i'%int(seg)] = alldata.data['JdotE%.8i'%int(seg)] + alldata.calc_scalr('J_x','VexB_x','J_y','VexB_y','J_z','VexB_z',seg=seg,fill=False)
104 for sp in alldata.meta['species']:
105 if tar=='Tiso' :
106 alldata.data[tar+sp+'%.8i'%int(seg)] = (alldata.data['TXX'+sp+'%.8i'%int(seg)]+alldata.data['TYY'+sp+'%.8i'%int(seg)]+alldata.data['TZZ'+sp+'%.8i'%int(seg)])/3.
107 if tar=='Tper' :
108 alldata.data['temp_x%.8i'%int(seg)] = alldata.calc_scalr('TXX'+sp,'B_x','TXY'+sp,'B_y','TXZ'+sp,'B_z',seg=seg,fill=False)
109 alldata.data['temp_y%.8i'%int(seg)] = alldata.calc_scalr('TXY'+sp,'B_x','TYY'+sp,'B_y','TYZ'+sp,'B_z',seg=seg,fill=False)
110 alldata.data['temp_z%.8i'%int(seg)] = alldata.calc_scalr('TXZ'+sp,'B_x','TYZ'+sp,'B_y','TZZ'+sp,'B_z',seg=seg,fill=False)
111 mod2B = alldata.calc_scalr('B_x','B_x','B_y','B_y','B_z','B_z',seg=seg,fill=False)
112 alldata.data[tar+sp+'%.8i'%int(seg)] = (3.*alldata.data['Tiso'+sp+'%.8i'%int(seg)]-(alldata.calc_scalr('temp_x','B_x','temp_y','B_y','temp_z','B_z',seg=seg,fill=False)/mod2B))
113 if tar=='Tpar' :
114 alldata.data[tar+sp+'%.8i'%int(seg)] = alldata.calc_scalr('temp_x','B_x','temp_y','B_y','temp_z','B_z',seg=seg,fill=False)/mod2B
115
116 print(alldata.data.keys())
117
118
119#----print-derived-fields-----------------------
120 print('PRINT->',seg)
121 for tar in out_tars:
122 if 'Jp' in tar :
123 alldata.print_vtk_vect(tar+'B_x',tar+'B_y',tar+'B_z',seg,data_address,alldata.meta['name']+'_'+tar+cut+'_'+str(seg),silent=True)
124 elif 'E' in tar :
125 alldata.print_vtk_scal(tar,seg,data_address,alldata.meta['name']+'_'+tar+cut+'_'+str(seg),silent=True)
126 else :
127 for sp in alldata.meta['species']:
128 alldata.print_vtk_scal(tar+sp,seg,data_address,alldata.meta['name']+'_'+tar+sp+cut+'_'+str(seg),silent=True)
129
130#----del-3D-fields----------------------------
131 print('DELETE->',seg)
132 for key in alldata.data.keys():
133 del alldata.data[key]
134
135#----print-comp-time--------------------
136 print('COMP.TIME [seg%i] = %.3f s\n'%(seg,time.time()-t0))
