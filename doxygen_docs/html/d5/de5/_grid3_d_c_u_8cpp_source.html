31/*! constructor */
32Grid3DCU::Grid3DCU(CollectiveIO * col, VirtualTopology3D * vct):
33 _vct(*vct)
34{
35 assert_le(col->getXLEN(),col->getNxc());
36 assert_le(col->getYLEN(),col->getNyc());
37 assert_le(col->getZLEN(),col->getNzc());
38
39 // get number of cells restricted to regular (untruncated) subdomain
40 const int nxc_rr = ceiling_of_ratio(col->getNxc(), col->getXLEN());
41 const int nyc_rr = ceiling_of_ratio(col->getNyc(), col->getYLEN());
42 const int nzc_rr = ceiling_of_ratio(col->getNzc(), col->getZLEN());
43
44 nxc_r = nxc_rr;
45 nyc_r = nyc_rr;
46 nzc_r = nzc_rr;
47
48 num_cells_rr = nxc_rr*nyc_rr*nzc_rr;
49
50 // Truncate the number of cells appropriately in the upper process.
51 // (We truncate rather than extend to avoid causing any load imbalance.)
52 //
53 if(vct->isXupper()) nxc_r = col->getNxc()-nxc_r*(col->getXLEN()-1);
54 if(vct->isYupper()) nyc_r = col->getNyc()-nyc_r*(col->getYLEN()-1);
55 if(vct->isZupper()) nzc_r = col->getNzc()-nzc_r*(col->getZLEN()-1);
56 //
57 assert_lt(0,nxc_r); assert_le(nxc_r,nxc_rr);
58 assert_lt(0,nyc_r); assert_le(nyc_r,nyc_rr);
59 assert_lt(0,nzc_r); assert_le(nzc_r,nzc_rr);
60
61 // These restrictions should be removed.
62 //
63 // An objection to removing them is that the user can always
64 // increase the number of mesh cells appropriately to be a
65 // multiple of XLEN, but as a rejoinder, it is desirable to be
66 // able to change the number of processors without changing
67 // the discretized problem in any way and without divisibility
68 // restrictions.
69 //
70 //assert_divides(vct->getXLEN(), col->getNxc());
71 //assert_divides(vct->getYLEN(), col->getNyc());
72 //assert_divides(vct->getZLEN(), col->getNzc());
73 //
74 //assert_eq(nxc_r,nxc_rr);
75 //assert_eq(nyc_r,nyc_rr);
76 //assert_eq(nzc_r,nzc_rr);
77
78 // add two for ghost cells
79 nxc = nxc_r + 2;
80 nyc = nyc_r + 2;
81 nzc = nzc_r + 2;
82
83 dx = col->getDx();
84 dy = col->getDy();
85 dz = col->getDz();
86 assert(dx == col->getLx() / col->getNxc());
87 assert(dy == col->getLy() / col->getNyc());
88 assert(dz == col->getLz() / col->getNzc());
89
90 // local grid dimensions and boundaries of active nodes
91 //
92 // width of an ordinary subdomain
93 //
94 const double xWidth = dx*nxc_rr;
95 const double yWidth = dy*nyc_rr;
96 const double zWidth = dz*nzc_rr;
97 //const double xWidth = (col->getLx() / (double) vct->getXLEN());
98 //const double yWidth = (col->getLy() / (double) vct->getYLEN());
99 //const double zWidth = (col->getLz() / (double) vct->getZLEN());
100 //assert_almost_eq(dx*(nxc_r),xWidth,dx*1e-8);
101 //assert_almost_eq(dy*(nyc_r),yWidth,dy*1e-8);
102 //assert_almost_eq(dz*(nzc_r),zWidth,dz*1e-8);
103 //
104 xStart = vct->getCoordinates(0) * xWidth;
105 yStart = vct->getCoordinates(1) * yWidth;
106 zStart = vct->getCoordinates(2) * zWidth;
107 //
108 xEnd = xStart + xWidth - (nxc_rr-nxc_r)*dx;
109 yEnd = yStart + yWidth - (nyc_rr-nyc_r)*dy;
110 zEnd = zStart + zWidth - (nzc_rr-nzc_r)*dz;
111
112 init_derived_parameters();
113}

134// set derived convenience parameters
135void Grid3DCU::init_derived_parameters()
136{
137 epsilon = (nxc+nyc+nzc)*1e-15;
138 nxc_minus_epsilon = nxc-epsilon;
139 nyc_minus_epsilon = nyc-epsilon;
140 nzc_minus_epsilon = nzc-epsilon;
141 assert_lt(int(floor(nxc_minus_epsilon)),nxc);
142 assert_lt(int(floor(nyc_minus_epsilon)),nyc);
143 assert_lt(int(floor(nzc_minus_epsilon)),nzc);
144 xStart_g = xStart - dx;
145 yStart_g = yStart - dy;
146 zStart_g = zStart - dz;
147 // calculation conveniences
148 //
149 VOL = dx * dy * dz;
150 invVOL = 1.0 / VOL;
151 invdx = 1.0 / dx;
152 invdy = 1.0 / dy;
153 invdz = 1.0 / dz;
154 //
155 nxn = nxc + 1;
156 nyn = nyc + 1;
157 nzn = nzc + 1;
158 cxlast = nxc-1;
159 cylast = nyc-1;
160 czlast = nzc-1;
161
162 // arrays allocation: nodes ---> the first node has index 1, the last has index nxn-2!
163 pfloat_node_xcoord = new pfloat[nxn];
164 pfloat_node_ycoord = new pfloat[nyn];
165 pfloat_node_zcoord = new pfloat[nzn];
166 node_xcoord = new double[nxn];
167 node_ycoord = new double[nyn];
168 node_zcoord = new double[nzn];
169 for (int i=0; i<nxn; i++) node_xcoord[i] = xStart + (i - 1) * dx;
170 for (int j=0; j<nyn; j++) node_ycoord[j] = yStart + (j - 1) * dy;
171 for (int k=0; k<nzn; k++) node_zcoord[k] = zStart + (k - 1) * dz;
172 for (int i=0; i<nxn; i++) pfloat_node_xcoord[i] = node_xcoord[i];
173 for (int j=0; j<nyn; j++) pfloat_node_ycoord[j] = node_ycoord[j];
174 for (int k=0; k<nzn; k++) pfloat_node_zcoord[k] = node_zcoord[k];
175 // arrays allocation: cells ---> the first cell has index 1, the last has index ncn-2!
176 center_xcoord = new double[nxc];
177 center_ycoord = new double[nyc];
178 center_zcoord = new double[nzc];
179 for(int i=0; i<nxc; i++) center_xcoord[i] = .5*(node_xcoord[i]+node_xcoord[i+1]);
180 for(int j=0; j<nyc; j++) center_ycoord[j] = .5*(node_ycoord[j]+node_ycoord[j+1]);
181 for(int k=0; k<nzc; k++) center_zcoord[k] = .5*(node_zcoord[k]+node_zcoord[k+1]);
182 //num_cells_r = nxc_r*nyc_r*nzc_r;
183 //num_cells = nxc*nyc*nzc;
184}

217/** calculate gradient on nodes, given a scalar field defined on central points */
218void Grid3DCU::gradC2N(arr3_double gradXN, arr3_double gradYN, arr3_double gradZN, const_arr3_double scFieldC)const
219{
220 for (register int i = 1; i < nxn - 1; i++)
221 for (register int j = 1; j < nyn - 1; j++)
222 for (register int k = 1; k < nzn - 1; k++) {
223 gradXN[i][j][k] = .25 * (scFieldC[i][j][k] - scFieldC[i - 1][j][k]) * invdx + .25 * (scFieldC[i][j][k - 1] - scFieldC[i - 1][j][k - 1]) * invdx + .25 * (scFieldC[i][j - 1][k] - scFieldC[i - 1][j - 1][k]) * invdx + .25 * (scFieldC[i][j - 1][k - 1] - scFieldC[i - 1][j - 1][k - 1]) * invdx;
224 gradYN[i][j][k] = .25 * (scFieldC[i][j][k] - scFieldC[i][j - 1][k]) * invdy + .25 * (scFieldC[i][j][k - 1] - scFieldC[i][j - 1][k - 1]) * invdy + .25 * (scFieldC[i - 1][j][k] - scFieldC[i - 1][j - 1][k]) * invdy + .25 * (scFieldC[i - 1][j][k - 1] - scFieldC[i - 1][j - 1][k - 1]) * invdy;
225 gradZN[i][j][k] = .25 * (scFieldC[i][j][k] - scFieldC[i][j][k - 1]) * invdz + .25 * (scFieldC[i - 1][j][k] - scFieldC[i - 1][j][k - 1]) * invdz + .25 * (scFieldC[i][j - 1][k] - scFieldC[i][j - 1][k - 1]) * invdz + .25 * (scFieldC[i - 1][j - 1][k] - scFieldC[i - 1][j - 1][k - 1]) * invdz;
226 }
227}

241/** calculate divergence on central points, given a vector field defined on nodes */
242void Grid3DCU::divN2C(arr3_double divC, const_arr3_double vecFieldXN, const_arr3_double vecFieldYN, const_arr3_double vecFieldZN)const
243{
244 double compX;
245 double compY;
246 double compZ;
247 for (register int i = 1; i < nxc - 1; i++)
248 for (register int j = 1; j < nyc - 1; j++)
249 for (register int k = 1; k < nzc - 1; k++) {
250 compX = .25 * (vecFieldXN[i + 1][j][k] - vecFieldXN[i][j][k]) * invdx + .25 * (vecFieldXN[i + 1][j][k + 1] - vecFieldXN[i][j][k + 1]) * invdx + .25 * (vecFieldXN[i + 1][j + 1][k] - vecFieldXN[i][j + 1][k]) * invdx + .25 * (vecFieldXN[i + 1][j + 1][k + 1] - vecFieldXN[i][j + 1][k + 1]) * invdx;
251 compY = .25 * (vecFieldYN[i][j + 1][k] - vecFieldYN[i][j][k]) * invdy + .25 * (vecFieldYN[i][j + 1][k + 1] - vecFieldYN[i][j][k + 1]) * invdy + .25 * (vecFieldYN[i + 1][j + 1][k] - vecFieldYN[i + 1][j][k]) * invdy + .25 * (vecFieldYN[i + 1][j + 1][k + 1] - vecFieldYN[i + 1][j][k + 1]) * invdy;
252 compZ = .25 * (vecFieldZN[i][j][k + 1] - vecFieldZN[i][j][k]) * invdz + .25 * (vecFieldZN[i + 1][j][k + 1] - vecFieldZN[i + 1][j][k]) * invdz + .25 * (vecFieldZN[i][j + 1][k + 1] - vecFieldZN[i][j + 1][k]) * invdz + .25 * (vecFieldZN[i + 1][j + 1][k + 1] - vecFieldZN[i + 1][j + 1][k]) * invdz;
253 divC[i][j][k] = compX + compY + compZ;
254 }

264 for (register int j = 1; j < nyc - 1; j++)
265 for (register int k = 1; k < nzc - 1; k++) {
266 comp1X = .25 * (pXX[ns][i + 1][j][k] - pXX[ns][i][j][k]) * invdx + .25 * (pXX[ns][i + 1][j][k + 1] - pXX[ns][i][j][k + 1]) * invdx + .25 * (pXX[ns][i + 1][j + 1][k] - pXX[ns][i][j + 1][k]) * invdx + .25 * (pXX[ns][i + 1][j + 1][k + 1] - pXX[ns][i][j + 1][k + 1]) * invdx;
267 comp2X = .25 * (pXY[ns][i + 1][j][k] - pXY[ns][i][j][k]) * invdx + .25 * (pXY[ns][i + 1][j][k + 1] - pXY[ns][i][j][k + 1]) * invdx + .25 * (pXY[ns][i + 1][j + 1][k] - pXY[ns][i][j + 1][k]) * invdx + .25 * (pXY[ns][i + 1][j + 1][k + 1] - pXY[ns][i][j + 1][k + 1]) * invdx;
268 comp3X = .25 * (pXZ[ns][i + 1][j][k] - pXZ[ns][i][j][k]) * invdx + .25 * (pXZ[ns][i + 1][j][k + 1] - pXZ[ns][i][j][k + 1]) * invdx + .25 * (pXZ[ns][i + 1][j + 1][k] - pXZ[ns][i][j + 1][k]) * invdx + .25 * (pXZ[ns][i + 1][j + 1][k + 1] - pXZ[ns][i][j + 1][k + 1]) * invdx;
269 comp1Y = .25 * (pXY[ns][i][j + 1][k] - pXY[ns][i][j][k]) * invdy + .25 * (pXY[ns][i][j + 1][k + 1] - pXY[ns][i][j][k + 1]) * invdy + .25 * (pXY[ns][i + 1][j + 1][k] - pXY[ns][i + 1][j][k]) * invdy + .25 * (pXY[ns][i + 1][j + 1][k + 1] - pXY[ns][i + 1][j][k + 1]) * invdy;
270 comp2Y = .25 * (pYY[ns][i][j + 1][k] - pYY[ns][i][j][k]) * invdy + .25 * (pYY[ns][i][j + 1][k + 1] - pYY[ns][i][j][k + 1]) * invdy + .25 * (pYY[ns][i + 1][j + 1][k] - pYY[ns][i + 1][j][k]) * invdy + .25 * (pYY[ns][i + 1][j + 1][k + 1] - pYY[ns][i + 1][j][k + 1]) * invdy;
271 comp3Y = .25 * (pYZ[ns][i][j + 1][k] - pYZ[ns][i][j][k]) * invdy + .25 * (pYZ[ns][i][j + 1][k + 1] - pYZ[ns][i][j][k + 1]) * invdy + .25 * (pYZ[ns][i + 1][j + 1][k] - pYZ[ns][i + 1][j][k]) * invdy + .25 * (pYZ[ns][i + 1][j + 1][k + 1] - pYZ[ns][i + 1][j][k + 1]) * invdy;
272 comp1Z = .25 * (pXZ[ns][i][j][k + 1] - pXZ[ns][i][j][k]) * invdz + .25 * (pXZ[ns][i + 1][j][k + 1] - pXZ[ns][i + 1][j][k]) * invdz + .25 * (pXZ[ns][i][j + 1][k + 1] - pXZ[ns][i][j + 1][k]) * invdz + .25 * (pXZ[ns][i + 1][j + 1][k + 1] - pXZ[ns][i + 1][j + 1][k]) * invdz;

274 comp3Z = .25 * (pZZ[ns][i][j][k + 1] - pZZ[ns][i][j][k]) * invdz + .25 * (pZZ[ns][i + 1][j][k + 1] - pZZ[ns][i + 1][j][k]) * invdz + .25 * (pZZ[ns][i][j + 1][k + 1] - pZZ[ns][i][j + 1][k]) * invdz + .25 * (pZZ[ns][i + 1][j + 1][k + 1] - pZZ[ns][i + 1][j + 1][k]) * invdz;
275 divCX[i][j][k] = comp1X + comp1Y + comp1Z;
276 divCY[i][j][k] = comp2X + comp2Y + comp2Z;
277 divCZ[i][j][k] = comp3X + comp3Y + comp3Z;

292 compZ = .25 * (vecFieldZC[i][j][k] - vecFieldZC[i][j][k - 1]) * invdz + .25 * (vecFieldZC[i - 1][j][k] - vecFieldZC[i - 1][j][k - 1]) * invdz + .25 * (vecFieldZC[i][j - 1][k] - vecFieldZC[i][j - 1][k - 1]) * invdz + .25 * (vecFieldZC[i - 1][j - 1][k] - vecFieldZC[i - 1][j - 1][k - 1]) * invdz;

294 }

297/** calculate curl on nodes, given a vector field defined on central points */
298void Grid3DCU::curlC2N(arr3_double curlXN, arr3_double curlYN, arr3_double curlZN, const_arr3_double vecFieldXC, const_arr3_double vecFieldYC, const_arr3_double vecFieldZC)const
299{
300 double compZDY, compYDZ;
301 double compXDZ, compZDX;
302 double compYDX, compXDY;
303 for (register int i = 1; i < nxn - 1; i++)
304 for (register int j = 1; j < nyn - 1; j++)
305 for (register int k = 1; k < nzn - 1; k++) {
306 // curl - X
307 compZDY = .25 * (vecFieldZC[i][j][k] - vecFieldZC[i][j - 1][k]) * invdy + .25 * (vecFieldZC[i][j][k - 1] - vecFieldZC[i][j - 1][k - 1]) * invdy + .25 * (vecFieldZC[i - 1][j][k] - vecFieldZC[i - 1][j - 1][k]) * invdy + .25 * (vecFieldZC[i - 1][j][k - 1] - vecFieldZC[i - 1][j - 1][k - 1]) * invdy;
308 compYDZ = .25 * (vecFieldYC[i][j][k] - vecFieldYC[i][j][k - 1]) * invdz + .25 * (vecFieldYC[i - 1][j][k] - vecFieldYC[i - 1][j][k - 1]) * invdz + .25 * (vecFieldYC[i][j - 1][k] - vecFieldYC[i][j - 1][k - 1]) * invdz + .25 * (vecFieldYC[i - 1][j - 1][k] - vecFieldYC[i - 1][j - 1][k - 1]) * invdz;
309 // curl - Y

377 array3_double gradYN(nxn, nyn, nzn);
378 array3_double gradZN(nxn, nyn, nzn);
379
380 gradC2N(gradXN, gradYN, gradZN, scFieldC);

383 for (int kk = 0; kk < nzn; kk++) {

394 for (int ii = 0; ii < nxn; ii++)
