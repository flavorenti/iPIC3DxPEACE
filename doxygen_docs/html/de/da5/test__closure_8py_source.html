1#--------------------------------------------------------------------
2# Code based on fibo library that (1) READS inp_tars, (2) SELECT the
3# values of these fields in a given region, (3) PRINT these values on .txt
4# The .txt are saved in the dir Closure_files in the same dir as python script.
5# Add global path to data dir in command line.
6# Only parameters to change are:
7#
8# inp_tars = [str] target variables you want to use (in order!)
9# limits = [list] limits to compute subset [[xmin,xmax],[ymin,ymax],[zmin,zmax]]
10# cuts = [str] cuts to use if the data are 2D (eg data1), if [''] data are 3D (eg data2)
11# cycle_min = [int] minimum cycle to read
12# cycle_max = [int] maximum cycle to read
13#
14# Extra rules: only scalars are admitted in inp_tars!
15#
16# Job 12/2021
17#-------------------------------------------------------------------
18
19######################################################################################
20#======regulate=parameters=============================================================
21inp_tars = ['E','Tiso','Tpar','Tper']
22limits = [[41,46],[30,36],[0.0,0.0]]
23cuts = ['eq']
24cycle_min = 0
25cycle_max = 2000000
26#=======================================================================================
27########################################################################################
28
29########################################################################################
30####################### DO NOT TOUCH BELOW! ############################################
31########################################################################################
32#----import-modules---------------
33import fibo as fb
34import time
35import sys
36import numpy as np
37
38#----path-to-run-------------------
39data_address = sys.argv[1]
40
41#----create-your-objects-----------
42fibo_name = 'ipic3d'
43alldata = fb.fibo(fibo_name)
44from_VTK = fb.from_VTK(data_address)
45
46#----loop-on-types-of-data--------
47for cut in cuts:
48
49#----load-metadata------------------
50 from_VTK.get_meta(silent=False)
51 if cut == 'eq':
52 from_VTK.meta['nz']=1
53 from_VTK.meta['nnn']=(from_VTK.meta['nx'],from_VTK.meta['ny'],1)
54 if cut == 'dp':
55 from_VTK.meta['ny']=1
56 from_VTK.meta['nnn']=(from_VTK.meta['nx'],1,from_VTK.meta['nz'])
57 alldata.meta = from_VTK.meta
58
59#----compute-limits---------------
60 ix1 = int(limits[0][0]/from_VTK.meta['dx'])
61 ix2 = int(limits[0][1]/from_VTK.meta['dx'])
62 iy1 = int(limits[1][0]/from_VTK.meta['dy'])
63 iy2 = int(limits[1][1]/from_VTK.meta['dy'])
64 iz1 = int(limits[2][0]/from_VTK.meta['dz'])
65 iz2 = int(limits[2][1]/from_VTK.meta['dz'])
66
67#----load-everything-in-code-units-------------------
68 for seg in alldata.meta['segcycles']:
69 t0 = time.time()
70 if cycle_min<=seg<=cycle_max :
71 print('LOAD->',seg)
72 for tar in inp_tars:
73 if len(tar)<3 :
74 from_VTK.get_vect(alldata.meta['name']+'_'+tar+cut+'_'+str(seg),seg,fibo_obj=alldata,tar_var=tar,silent=True)
75 else:
76 for sp in alldata.meta['species']:
77 from_VTK.get_scal(alldata.meta['name']+'_'+tar+sp+cut+'_'+str(seg),seg,fibo_obj=alldata,tar_var=tar+sp,silent=True)
78
79 print(alldata.data.keys())
80
81#----compute-derived-fields------------------
82 print('SELECT+PRINT->',seg)
83 for tar in inp_tars:
84 if len(tar)<3 :
85 file_out = open('closure_files/closure_'+tar+cut+'.txt','a')
86 if int(seg)==0 : file_out.write('#seg\t value\n')
87 if ix1>=ix2 :
88 out = np.ravel(alldata.data[tar+'_y%.8i'%int(seg)][:,iy1:iy2,iz1:iz2])
89 if iy1>=iy2 :
90 out = np.ravel(alldata.data[tar+'_y%.8i'%int(seg)][ix1:ix2,:,iz1:iz2])
91 if iz1>=iz2 :
92 out = np.ravel(alldata.data[tar+'_y%.8i'%int(seg)][ix1:ix2,iy1:iy2,:])
93 out = out[~np.isnan(out)]
94 out = out[~np.isinf(out)]
95 file_out.write('%i\t%.9f\n'%(int(seg),np.mean(out)))
96 file_out.close()
97
98 else:
99 for sp in alldata.meta['species']:
100 file_out = open('closure_files/closure_'+tar+sp+cut+'.txt','a')
101 if int(seg)==0 : file_out.write('#seg\t value\n')
102 if ix1>=ix2 :
103 out = np.ravel(alldata.data[tar+sp+'%.8i'%int(seg)][:,iy1:iy2,iz1:iz2])
104 if iy1>=iy2 :
105 out = np.ravel(alldata.data[tar+sp+'%.8i'%int(seg)][ix1:ix2,:,iz1:iz2])
106 if iz1>=iz2 :
107 out = np.ravel(alldata.data[tar+sp+'%.8i'%int(seg)][ix1:ix2,iy1:iy2,:])
108 out = out[~np.isnan(out)]
109 out = out[~np.isinf(out)]
110 file_out.write('%i\t%.9f\n'%(int(seg),np.mean(out)))
111 file_out.close()
112
113#----del-3D-fields----------------------------
114 print('DELETE->',seg)
115 for key in alldata.data.keys():
116 del alldata.data[key]
117
118#----print-comp-time--------------------
119 print('COMP.TIME [seg%i] = %.3f s\n'%(seg,time.time()-t0))
