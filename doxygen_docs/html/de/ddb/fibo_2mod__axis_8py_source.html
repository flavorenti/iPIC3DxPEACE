19 def axis_data(self,
20 range_x = None,

29 wrap_like = False):
30 """
31 Calculates axes values for some range around a given origin
32
33 Parameters :
34 - range_x = None [int,int] x coordinates of the first and last point
35 - range_y = None [int,int] y coordinates of the first and last point
36 - range_z = None [int,int] z coordinates of the first and last point
37 - offset_x = 0 [int] x coordinate of the origin
38 - offset_y = 0 [int] y coordinate of the origin
39 - offset_z = 0 [int] z coordinate of the origin
40 - coor_like = False [bool] do you want axes to be coordinate-like scaled?
41 - squa_like = False [bool] do you want axes to measure square of distance from origin?
42 - mesh_like = False [bool] do you want axes in mesh mode?
43
44 Returns :
45 - x_coor [fibo.data OR np.ndarray(nx)]
46 - y_coor [fibo.data OR np.ndarray(ny)]
47 - z_coor [fibo.data OR np.ndarray(nz)]
48
49 """
50
51 nx,ny,nz = self.meta['nnn']
52
53 if range_x is None : range_x = [0,nx]
54 if range_y is None : range_y = [0,ny]
55 if range_z is None : range_z = [0,nz]
56
57 # here I determine how many relevant periodic dimensions there are
58 dim_x = self.meta['ppp'][0] and self.meta['nnn'][0] != 1
59 dim_y = self.meta['ppp'][1] and self.meta['nnn'][1] != 1
60 dim_z = self.meta['ppp'][2] and self.meta['nnn'][2] != 1
61
62 # for every relevant periodic dimension, change the range if necessary
63 if dim_x : range_x[0] = (range_x[0] + nx)%nx
64 if dim_y : range_y[0] = (range_y[0] + ny)%ny
65 if dim_z : range_z[0] = (range_z[0] + nz)%nz
66
67 if dim_x : range_x[1] = range_x[0]+(range_x[1]-range_x[0])%(nx+1)
68 if dim_y : range_y[1] = range_y[0]+(range_y[1]-range_y[0])%(ny+1)
69 if dim_z : range_z[1] = range_z[0]+(range_z[1]-range_z[0])%(nz+1)
70
71 # now let's produce the first version of our output arrays
72 x_coor = np.arange( range_x[0], range_x[1] ) - offset_x
73 y_coor = np.arange( range_y[0], range_y[1] ) - offset_y
74 z_coor = np.arange( range_z[0], range_z[1] ) - offset_z
75
76 # and re-change them over every periodic dimension
77 if wrap_like :
78 if dim_x : x_coor = x_coor%nx
79 if dim_y : y_coor = y_coor%ny
80 if dim_z : z_coor = z_coor%nz
81
82 # if you want coor-like: multiply by dx,dy,dz
83 if coor_like :
84 x_coor = x_coor*self.meta['ddd'][0]
85 y_coor = y_coor*self.meta['ddd'][1]
86 z_coor = z_coor*self.meta['ddd'][2]
87
88 # if you want these squared (btw what? - whatever ...)
89 if squa_like :
90 x_coor = np.square(x_coor)
91 y_coor = np.square(y_coor)
92 z_coor = np.square(z_coor)
93
94 # and if you want the mesh also (notice the weird order - yeah)
95 if mesh_like :
96 y_coor, x_coor, z_coor = np.meshgrid(y_coor,x_coor,z_coor)
97
98
99 return x_coor, y_coor, z_coor

108 Calculates axis values along a specified line
109
110 Parameters :

113 - range_y [int,int] y coordinates of the first and last point
114 - range_z [int,int] z coordinates of the first and last point
115
116 Returns :
117 - line [np.ndarray(line_pts)]
118
119 """
120
121 line_len = (self.meta['dx']*(range_x[1]-range_x[0]))**2
122 line_len += (self.meta['dy']*(range_y[1]-range_y[0]))**2
123 line_len += (self.meta['dz']*(range_z[1]-range_z[0]))**2
124 line_len = np.sqrt(line_len)
125
126 return np.linspace(0, line_len, line_pts, endpoint=True)
