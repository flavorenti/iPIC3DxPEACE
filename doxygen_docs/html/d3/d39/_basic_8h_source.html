172/** method to calculate the epsilon machine */
173inline double eps() {
174 double eps;
175 int i = 1;
176 double num = 1;
177 double newsum = 1;
178 double oldsum = 1;
179 while (true) {
180 num = num / (2 * i);
181 newsum += num;
182 if (newsum == oldsum)
183 break;
184 oldsum = newsum;
185 i++;
186 }
187 eps = num * 2;
188 return (eps);
189}
