2# Code based on fibo library that (1) READS, (2) CUT and (3) PRINT
3# the three-dimensional output of an iPIC3D simu. The cuts are done
4# in Lz/2 and Ly/2. The cuts are saved in the same dir as the data,
5# named as e.g. Beq, Bdp for equatorial and dipolar cut.
6# Add global path to data dir in command line.
7# Only parameters to change are:
8#
9# tars = [str] target variables you want to use (in order!)
10# cycle_min = [int] minimum cycle to read
11# cycle_max = [int] maximum cycle to read
12# diffRho = [bool] do you want to compute total charge rhoi+rhoe?
13#
14# Job 11/2021
15#-------------------------------------------------------------------
16
17######################################################################################
18#======regulate=parameters=============================================================
19tars = ['B','E','Je','Ji','PXX','PXY','PXZ','PYY','PYZ','PZZ','rho']
20cycle_min = 0
21cycle_max = 200000
22offset = 0
23diffRho = True

30#----import-modules---------------
31import fibo as fb
32import time
33import sys
34import numpy as np
35from scipy.ndimage.filters import gaussian_filter as gf
36
37#----path-to-run-------------------
38data_address = sys.argv[1]
39
40#----create-your-objects-----------
41fibo_name = 'ipic3d'
42data3D = fb.fibo(fibo_name)
43from_VTK = fb.from_VTK(data_address)
44
45#----load-metadata------------------
46from_VTK.get_meta(silent=False)
47data3D.meta = from_VTK.meta
48nx,ny,nz = from_VTK.meta['nnn']
49
50#---define-cuts-you-want----------
51cuts = []
52fibos = []
53cuts.append(['eq',[0,nx],[0,ny],[int(nz/2),int(nz/2)+1]])
54cuts.append(['dp',[0,nx],[int(ny/2),int(ny/2)+1],[0,nz]])
55for cut in cuts:
56 data2D = fb.fibo(fibo_name)
57 data2D.meta = from_VTK.meta
58 fibos.append(data2D)
59
60#----load-everything-in-code-units-------------------
61for seg in data3D.meta['segcycles']:
62 seg=seg+offset
63 t0 = time.time()
64 if cycle_min<=seg<=cycle_max :
65 print('LOAD->',seg)
66 for tar in tars:
67 if (('P' in tar) or ('rho' in tar)) :
68 for sp in data3D.meta['species']:
69 from_VTK.get_scal(data3D.meta['name']+'_'+tar+sp+'_'+str(seg),seg,fibo_obj=data3D,tar_var=tar+sp,silent=True)
70 else:
71 from_VTK.get_vect(data3D.meta['name']+'_'+tar+'_'+str(seg),seg,fibo_obj=data3D,tar_var=tar,silent=True)
72
73#----cut-to-2D------------------
74 print('CUT->',seg)
75 for ic,cut in enumerate(cuts):
76 for tar in tars:
77 if (('P' in tar) or ('rho' in tar)) :
78 for sp in data3D.meta['species']:
79 fibos[ic].data[tar+sp+'%.8i'%int(seg)], fibos[ic].meta = data3D.extract_range(tar+sp+'%.8i'%int(seg),cut[1],cut[2],cut[3])
80 else:
81 fibos[ic].data[tar+'_x%.8i'%int(seg)], fibos[ic].meta = data3D.extract_range(tar+'_x%.8i'%int(seg),cut[1],cut[2],cut[3])
82 fibos[ic].data[tar+'_y%.8i'%int(seg)], fibos[ic].meta = data3D.extract_range(tar+'_y%.8i'%int(seg),cut[1],cut[2],cut[3])
83 fibos[ic].data[tar+'_z%.8i'%int(seg)], fibos[ic].meta = data3D.extract_range(tar+'_z%.8i'%int(seg),cut[1],cut[2],cut[3])
84 if diffRho :
85 fibos[ic].data['Drho%.8i'%int(seg)] = np.zeros(np.shape(fibos[ic].data['rhoi1%.8i'%int(seg)][:,:,:]))
86 for isp,sp in enumerate(data3D.meta['species']):
87 fibos[ic].data['Drho%.8i'%int(seg)] += fibos[ic].data['rho'+sp+'%.8i'%int(seg)]
88 fibos[ic].data['Drho%.8i'%int(seg)] = gf(fibos[ic].data['Drho%.8i'%int(seg)],3)
89
90#----print-derived-fields-----------------------
91 print('PRINT->',seg)
92 for ic,cut in enumerate(cuts):
93 for tar in tars:
94 if (('P' in tar) or ('rho' in tar)) :
95 for sp in data3D.meta['species']:
96 fibos[ic].print_vtk_scal(tar+sp,seg,data_address,data3D.meta['name']+'_'+tar+sp+cut[0]+'_'+str(seg),silent=True)
97 else:
98 fibos[ic].print_vtk_vect(tar+'_x',tar+'_y',tar+'_z',seg,data_address,data3D.meta['name']+'_'+tar+cut[0]+'_'+str(seg),silent=True)
99 if diffRho :
100 fibos[ic].print_vtk_scal('Drho',seg,data_address,data3D.meta['name']+'_Drho'+cut[0]+'_'+str(seg),silent=True)
101
102#----del-3D-fields----------------------------
103 print('DELETE->',seg)
104 for ic,cut in enumerate(cuts):
105 for key in fibos[ic].data.keys():
106 del fibos[ic].data[key]
107
108#----print-comp-time--------------------
109 print('COMP.TIME [seg%i] = %.3f s\n'%(seg,time.time()-t0))
