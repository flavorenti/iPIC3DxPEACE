32/** DEFINE THE Topology HERE, setting XLEN,YLEN,ZLEN */
33VCtopology3D::VCtopology3D(const Collective& col) {
34 // *******************************************
35 // *******************************************
36 // change these values to change the topology
37 XLEN = col.getXLEN();
38 YLEN = col.getYLEN();
39 ZLEN = col.getZLEN();
40 nprocs = XLEN * YLEN * ZLEN;
41 // here you have to set the topology
42 PERIODICX = col.getPERIODICX();
43 PERIODICY = col.getPERIODICY();
44 PERIODICZ = col.getPERIODICZ();
45
46 PERIODICX_P = col.getPERIODICX_P();
47 PERIODICY_P = col.getPERIODICY_P();
48 PERIODICZ_P = col.getPERIODICZ_P();
49
50 // *******************************************
51 // *******************************************
52 XDIR = 0;
53 YDIR = 1;
54 ZDIR = 2;
55 RIGHT = 1;
56 LEFT = -1;
57
58 reorder = 1;
59
60 dims[0] = XLEN;
61 dims[1] = YLEN;
62 dims[2] = ZLEN;
63
64 periods[0] = PERIODICX;
65 periods[1] = PERIODICY;
66 periods[2] = PERIODICZ;
67
68 periods_P[0] = PERIODICX_P;
69 periods_P[1] = PERIODICY_P;
70 periods_P[2] = PERIODICZ_P;
71
72 cVERBOSE = false; // communication verbose ?

74}

77/** Within CART_COMM, processes find about their new rank numbers, their cartesian coordinates,
78 and their neighbors */
79void VCtopology3D::setup_vctopology(MPI_Comm old_comm) {
80 // create a matrix with ranks, and neighbours for fields
81 MPI_Cart_create(old_comm, 3, dims, periods, reorder, &CART_COMM);
82
83 // create a matrix with ranks, and neighbours for Particles
84 MPI_Cart_create(old_comm, 3, dims, periods_P, reorder, &CART_COMM_P);
85
86 // field Communicator
87 if (CART_COMM != MPI_COMM_NULL) {
88 MPI_Comm_rank(CART_COMM, &cartesian_rank);
89 MPI_Cart_coords(CART_COMM, cartesian_rank, 3, coordinates);
90
91 MPI_Cart_shift(CART_COMM, XDIR, RIGHT, &xleft_neighbor, &xright_neighbor);
92 MPI_Cart_shift(CART_COMM, YDIR, RIGHT, &yleft_neighbor, &yright_neighbor);
93 MPI_Cart_shift(CART_COMM, ZDIR, RIGHT, &zleft_neighbor, &zright_neighbor);
94
95 }
96 else {
97 // previous check that nprocs = XLEN*YLEN*ZLEN should prevent reaching this line.
98 eprintf("A process is thrown away from the new topology for fields.");
99 }
100 // Particles Communicator
101 //if (CART_COMM_P != MPI_COMM_NULL) {
102 // int pcl_coordinates[3];
103 // int pcl_cartesian_rank;
104 // MPI_Comm_rank(CART_COMM_P, &pcl_cartesian_rank);
105 // MPI_Cart_coords(CART_COMM_P, pcl_cartesian_rank, 3, pcl_coordinates);
106 //
107 // This seems to be assumed elsewhere in the code.
108 // We need to eliminate this assumption.
109 // This becomes important if we want
110 // a running program to have more than one
111 // mesh and processor topology.
112 // The MPI rank is for system-level code, e.g.
113 // to identify the process from which debug is coming.
114 // The cartesian rank is for application-level code.
115 assert_eq(cartesian_rank, MPIdata::get_rank());
116
117
118 if (CART_COMM_P != MPI_COMM_NULL) {
119 MPI_Cart_shift(CART_COMM_P, XDIR, RIGHT, &xleft_neighbor_P, &xright_neighbor_P);
120 MPI_Cart_shift(CART_COMM_P, YDIR, RIGHT, &yleft_neighbor_P, &yright_neighbor_P);
121 MPI_Cart_shift(CART_COMM_P, ZDIR, RIGHT, &zleft_neighbor_P, &zright_neighbor_P);
122 }
123
124 _isPeriodicXlower_P = PERIODICX_P && (coordinates[0]==0);
125 _isPeriodicXupper_P = PERIODICX_P && (coordinates[0]==dims[0]-1);
126 _isPeriodicYlower_P = PERIODICY_P && (coordinates[1]==0);
127 _isPeriodicYupper_P = PERIODICY_P && (coordinates[1]==dims[1]-1);
128 _isPeriodicZlower_P = PERIODICZ_P && (coordinates[2]==0);
129 _isPeriodicZupper_P = PERIODICZ_P && (coordinates[2]==dims[2]-1);
130
131 _noXleftNeighbor_P = (xleft_neighbor_P == MPI_PROC_NULL);
132 _noXrghtNeighbor_P = (xright_neighbor_P == MPI_PROC_NULL);
133 _noYleftNeighbor_P = (yleft_neighbor_P == MPI_PROC_NULL);
134 _noYrghtNeighbor_P = (yright_neighbor_P == MPI_PROC_NULL);
135 _noZleftNeighbor_P = (zleft_neighbor_P == MPI_PROC_NULL);
136 _noZrghtNeighbor_P = (zright_neighbor_P == MPI_PROC_NULL);
137
138 _isBoundaryProcess_P =
139 _noXleftNeighbor_P ||
140 _noXrghtNeighbor_P ||
141 _noYleftNeighbor_P ||
142 _noYrghtNeighbor_P ||
143 _noZleftNeighbor_P ||
144 _noZrghtNeighbor_P;
145
146}
