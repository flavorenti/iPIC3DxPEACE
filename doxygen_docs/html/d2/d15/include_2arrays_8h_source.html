32/*** begin FixedArray classes for use when dimensions are known at compile time. ***/
33//
34// These classes improve upon native fixed arrays as follows:
35// - bounds-checking is performed if CHECK_BOUNDS is defined,
36// - myarray(i,j) access is supported, and
37// - functions can return fixed-dimension arrays,
38// whereas since the C standard does not allow
39// one to return an array with fixed dimensions.
40//
41// The purpose of implementing these extensions is so that
42// fixed-dimension arrays can be used in iPic3D if doing so
43// yields a significant performance benefit for the choice of
44// architecture and compiler.
45
46template <class type, size_t s1>
47class FixedArray1D
48{
49 public:
50 type arr[s1];
51 public:
52 type& fetch(size_t n1)
53 {
54 check_bounds(n1,s1);
55 return arr[n1];
56 }
57 type& operator[](size_t n1)
58 {
59 check_bounds(n1,s1);
60 return arr[n1];
61 }
62};

64// auxiliary class for chained operator[] dereferencing of FixedArray2D
65//
66template <class type, size_t s1, size_t s2>
67class FixedArray2D1
68{
69 type (&arr)[s1][s2];
70 size_t n1;
71 public:
72 FixedArray2D1(type (&_arr)[s1][s2], size_t _n1) :
73 arr(_arr), n1(_n1) {};
74 type& operator[](size_t n2)
75 {
76 check_bounds(n1,s1);
77 check_bounds(n2,s2);
78 return arr[n1][n2];
79 }
80};

82template <class type, size_t s1, size_t s2>
83class FixedArray2D
84{
85 public:
86 type arr [s1][s2];
87 public:
88 type& fetch(size_t n1, size_t n2)
89 {
90 check_bounds(n1,s1);
91 check_bounds(n2,s2);
92 return arr[n1][n2];
93 }
94 // Chaining operator[] this way
95 // does not allow bounds checking
96 // and does not work beyond 2D.
97 //type* operator[](size_t n1) { return arr[n1]; }
98 FixedArray2D1<type,s1,s2> operator[](size_t n1)
99 { return FixedArray2D1<type,s1,s2>(arr,n1); }
100};
