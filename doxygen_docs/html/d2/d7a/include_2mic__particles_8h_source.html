28 inline F64vec8 sample_field_mic(
29 F64vec8 weights,
30 F64vec8 field_components[8])
31 {
32 F64vec8 fields = F64vec8(0.);
33 #pragma unroll
34 for(int c=0; c<8; c++)
35 fields += F64vec8(weights[c])*field_components[c];
36 return fields;
37 }

51 // construct weights for two particles based on their canonical positions
52 //
53 // each particle has 8 weights, one for each corner of its cell
54 //
55 inline void construct_weights_for_2pcls(F64vec8 weights[2], F64vec8 X0)
56 {
57 const F64vec8 X1 = F64vec8(1.) - X0;
58
59 // construct weights for particles
60 //
61 // need versions with broadcasted component.
62 //
63 // doing via swizzle works only for the x component and
64 // requires the same number of operations as the more general
65 // technique, so for simplicity I do them all the same rather
66 // than handling the x component with the commented-out lines.
67 //const F64vec8 Xx0 = X0.aaaa();
68 //const F64vec8 Xx1 = X1.aaaa();
69 const __mmask8 rmask_01010101 = _mm512_int2mask(0xaa); // mask=10101010
70 const __mmask8 rmask_00110011 = _mm512_int2mask(0xcc); // mask=11001100
71 const __mmask8 rmask_00001111 = _mm512_int2mask(0xf0); // mask=11110000
72 // construct weights for first particle
73 //
74 // weights for first particle use low vectors of X0 and X1:
75 //
76 // weights[0][0] = X0[0]*X0[1]*X0[2]; // weight000
77 // weights[0][1] = X0[0]*X0[1]*X1[2]; // weight001
78 // weights[0][2] = X0[0]*X1[1]*X0[2]; // weight010
79 // weights[0][3] = X0[0]*X1[1]*X1[2]; // weight011
80 // weights[0][4] = X1[0]*X0[1]*X0[2]; // weight100
81 // weights[0][5] = X1[0]*X0[1]*X1[2]; // weight101
82 // weights[0][6] = X1[0]*X1[1]*X0[2]; // weight110
83 // weights[0][7] = X1[0]*X1[1]*X1[2]; // weight111
84 //
85 {
86 //wx = cat_low_halves(Xx0,Xx1);
87 const F64vec8 wx = broadcast_mask_blend(X0[0],X1[0],rmask_00001111);
88 const F64vec8 wy = broadcast_mask_blend(X0[1],X1[1],rmask_00110011);
89 const F64vec8 wz = broadcast_mask_blend(X0[2],X1[2],rmask_01010101);
90 weights[0] = wx*wy*wz;
91 }
92 // construct weights for second particle
93 //
94 // weights for second particle use high vectors of X0 and X1:
95 //
96 // weights[1][0] = X0[4]*X0[5]*X0[6]; // weight000
97 // weights[1][1] = X0[4]*X0[5]*X1[6]; // weight001
98 // weights[1][2] = X0[4]*X1[5]*X0[6]; // weight010
99 // weights[1][3] = X0[4]*X1[5]*X1[6]; // weight011
100 // weights[1][4] = X1[4]*X0[5]*X0[6]; // weight100
101 // weights[1][5] = X1[4]*X0[5]*X1[6]; // weight101
102 // weights[1][6] = X1[4]*X1[5]*X0[6]; // weight110
103 // weights[1][7] = X1[4]*X1[5]*X1[6]; // weight111
104 //
105 {
106 //wx = cat_hgh_halves(Xx0,Xx1);
107 const F64vec8 wx = broadcast_mask_blend(X0[4],X1[4],rmask_00001111);
108 const F64vec8 wy = broadcast_mask_blend(X0[5],X1[5],rmask_00110011);
109 const F64vec8 wz = broadcast_mask_blend(X0[6],X1[6],rmask_01010101);
110 weights[1] = wx*wy*wz;
111 }
112 }

114 inline F64vec8 compute_uvg_for_2pcls(F64vec8 uorig, F64vec8 B, F64vec8 E, F64vec8 qdto2mc)
115 {
116 /* F64vec8::aaaa() etc is (defectively) not declared const */
117 /*const*/ F64vec8 Om = qdto2mc*B;
118 const F64vec8 Omx = Om.aaaa();
119 const F64vec8 Omy = Om.bbbb();
120 const F64vec8 Omz = Om.cccc();
121 // if pushing more than 2 particles at a time, could make
122 // use of reduction intrinsics (_mm512_mask_reduce_add_pd)
123 // and could calculate the reciprocal more efficiently.
124 const F64vec8 omsq_p1 = F64vec8(1.) + Omx*Omx + Omy*Omy + Omz*Omz;
125 // is there a faster way to implement the reciprocal?
126 const F64vec8 denom = reciprocal_F64vec8(omsq_p1);
127 // solve the position equation
128 /*const*/ F64vec8 ut = uorig + qdto2mc*E;
129 const F64vec8 udotOm =
130 ut.aaaa()*Omx
131 + ut.bbbb()*Omy
132 + ut.cccc()*Omz;
133
134 // solve the velocity equation
135 //
136 return (ut + cross_product(ut,Om)+udotOm*Om)*denom;
137 }

139 inline void get_field_components_for_cell(
140 F64vec8 field_components0[8],
141 F64vec8 field_components1[8],
142 const_arr4_double fieldForPcls,
143 I32vec16 cx_)
144 {
145 // interface to the right of cell
146 const I32vec16 ix_ = cx_ + I32vec16(1);
147
148 int* cx = (int*)&cx_;
149 int* ix = (int*)&ix_;
150
151 // compute field_components array for particles
152 //
153 // compute the 1-dimensional subscripts for
154 // the 8 corners of the mesh cell
155 //
156 {
157 // broadcast coordinates
158 const I32vec16 iX0 = I32vec16(ix[0]);
159 const I32vec16 iY0 = I32vec16(ix[1]);
160 const I32vec16 iZ0 = I32vec16(ix[2]);
161 const I32vec16 cX0 = I32vec16(cx[0]);
162 const I32vec16 cY0 = I32vec16(cx[1]);
163 const I32vec16 cZ0 = I32vec16(cx[2]);
164 const I32vec16 iX1 = I32vec16(ix[0+4]);
165 const I32vec16 iY1 = I32vec16(ix[1+4]);
166 const I32vec16 iZ1 = I32vec16(ix[2+4]);
167 const I32vec16 cX1 = I32vec16(cx[0+4]);
168 const I32vec16 cY1 = I32vec16(cx[1+4]);
169 const I32vec16 cZ1 = I32vec16(cx[2+4]);
170 // for each particle we want:
171 //
172 // field_components[0] = fieldForPcls[iX][iY][iZ]; // field000
173 // field_components[1] = fieldForPcls[iX][iY][cZ]; // field001
174 // field_components[2] = fieldForPcls[iX][cY][iZ]; // field010
175 // field_components[3] = fieldForPcls[iX][cY][cZ]; // field011
176 // field_components[4] = fieldForPcls[cX][iY][iZ]; // field100
177 // field_components[5] = fieldForPcls[cX][iY][cZ]; // field101
178 // field_components[6] = fieldForPcls[cX][cY][iZ]; // field110
179 // field_components[7] = fieldForPcls[cX][cY][cZ]; // field111
180 //
181 // broadcast coordinates of first and second particle to low and high end of I32vec16
182 const __mmask16 rmask_00ff = _mm512_int2mask(0xff00); // mask=1111111100000000
183 const I32vec16 iX = _mm512_mask_blend_epi32(rmask_00ff,iX0,iX1);
184 const I32vec16 iY = _mm512_mask_blend_epi32(rmask_00ff,iY0,iY1);
185 const I32vec16 iZ = _mm512_mask_blend_epi32(rmask_00ff,iZ0,iZ1);
186 const I32vec16 cX = _mm512_mask_blend_epi32(rmask_00ff,cX0,cX1);
187 const I32vec16 cY = _mm512_mask_blend_epi32(rmask_00ff,cY0,cY1);
188 const I32vec16 cZ = _mm512_mask_blend_epi32(rmask_00ff,cZ0,cZ1);
189 // now use masks to form all possible combinations of coordinates
190 const __mmask16 rmask_00001111x2 = _mm512_int2mask(0xf0f0); // mask=11110000x2
191 const __mmask16 rmask_00110011x2 = _mm512_int2mask(0xcccc); // mask=11001100x2
192 const __mmask16 rmask_01010101x2 = _mm512_int2mask(0xaaaa); // mask=10101010x2
193 // x, y, and z subscripts of the 8 corners for both particles
194 const I32vec16 sX = _mm512_mask_blend_epi32(rmask_00001111x2,iX,cX);
195 const I32vec16 sY = _mm512_mask_blend_epi32(rmask_00110011x2,iY,cY);
196 const I32vec16 sZ = _mm512_mask_blend_epi32(rmask_01010101x2,iZ,cZ);
197 // compute the starting 1-dimensional index for the 8 corners
198 const int nxn = fieldForPcls.dim1();
199 const int nyn = fieldForPcls.dim2();
200 const int nzn = fieldForPcls.dim3();
201 const int nnn = fieldForPcls.dim4();
202 const I32vec16 nXn(nxn);
203 const I32vec16 nYn(nyn);
204 const I32vec16 nZn(nzn);
205 // compiler could see whether this would fail, so should cost nothing
206 assert(nnn==8); // needed for field_components to be aligned
207 const I32vec16 nNn(nnn);
208 // unfortunately I32vec16 does not support this syntax,
209 //const MyI32vec16 subs = ((sX*nYn+sY)*nZn+sZ)*nNn;
210 const I32vec16 s1 = fused_multiply_add(sX,nYn,sY);
211 const I32vec16 s2 = fused_multiply_add(s1,nZn,sZ);
212 const I32vec16 subs = multiply(s2,nNn);
213 //printexpr(subs);
214 int const*const subs_arr = (int*)&subs;
215 // access underlying 1D array of fieldForPcls
216 const double* fieldForPcls1d = fieldForPcls.get_arr();
217 #pragma unroll
218 for(int i=0; i<8; i++)
219 {
220 field_components0[i] = *(F64vec8*)&fieldForPcls1d[subs_arr[i]];
221 }
222 #pragma unroll
223 for(int i=0; i<8; i++)
224 {
225 field_components1[i] = *(F64vec8*)&fieldForPcls1d[subs_arr[i+8]];
226 }
227 }
228 }
