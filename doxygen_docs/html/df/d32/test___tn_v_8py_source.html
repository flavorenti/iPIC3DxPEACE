2# Code based on fibo library that (1) READS, (2) COMPUTE n,T,V and (3) PRINT
3# the derived two-dimensional and three-dimensional output of an iPIC3D simu.
4# The computation starts form the pristine output of the simu (either 2D or 3D)
5# The new fields are saved in the same dir as the data, with the same dimensions,
6# named as follows PXXi1 -> TXXi1, rhoi1 -> ni1, Ji -> Vi etc.
7# Add global path to data dir in command line.
8# Only parameters to change are:
9#
10# inp_tars = [str] target variables you want to use as input (in order!)
11# out_tars = [str] name variables you want to create as output (same order)
12# cuts = [str] cuts to use if the data are 2D (eg data1), if [''] data are 3D (eg data2)
13# cycle_min = [int] minimum cycle to read
14# cycle_max = [int] maximum cycle to read
15#
16# Job 12/2021
17#-------------------------------------------------------------------
18
19######################################################################################
20#======regulate=parameters=============================================================
21inp_tars = ['Je','Ji','PXX','PXY','PXZ','PYY','PYZ','PZZ','rho']
22out_tars = ['Ve','Vi','TXX','TXY','TXZ','TYY','TYZ','TZZ']
23cuts = ['eq','dp']
24cycle_min = 0
25cycle_max = 2000000
26#=======================================================================================
27########################################################################################
28
29########################################################################################
30####################### DO NOT TOUCH BELOW! ############################################
31########################################################################################
32#----import-modules---------------
33import fibo as fb
34import time
35import sys
36import numpy as np
37from scipy.ndimage.filters import gaussian_filter as gf
38
39#----path-to-run-------------------
40data_address = sys.argv[1]
41
42#----create-your-objects-----------
43fibo_name = 'ipic3d'
44alldata = fb.fibo(fibo_name)
45from_VTK = fb.from_VTK(data_address)
46
47#----loop-on-types-of-data--------
48for cut in cuts:
49
50 #----load-metadata------------------
51 from_VTK.get_meta(silent=False)
52 if cut == 'eq':
53 from_VTK.meta['nz']=1
54 from_VTK.meta['nnn']=(from_VTK.meta['nx'],from_VTK.meta['ny'],1)
55 if cut == 'dp':
56 from_VTK.meta['ny']=1
57 from_VTK.meta['nnn']=(from_VTK.meta['nx'],1,from_VTK.meta['nz'])
58 alldata.meta = from_VTK.meta
59
60 #----load-everything-in-code-units-------------------
61 for seg in alldata.meta['segcycles']:
62 t0 = time.time()
63 if cycle_min<=seg<=cycle_max :
64 print('LOAD->',seg)
65 for tar in inp_tars:
66 if len(tar)<3 :
67 from_VTK.get_vect(alldata.meta['name']+'_'+tar+cut+'_'+str(seg),seg,fibo_obj=alldata,tar_var=tar,silent=True)
68 else :
69 for sp in alldata.meta['species']:
70 from_VTK.get_scal(alldata.meta['name']+'_'+tar+sp+cut+'_'+str(seg),seg,fibo_obj=alldata,tar_var=tar+sp,silent=True)
71
72 print(alldata.data.keys())
73
74#----change-units-&-smooth---------------------------
75 alldata.calc_units('iPIC', fibo_obj=alldata, silent=False)
76 for tar in inp_tars:
77 for sp in alldata.meta['species']:
78 if 'rho' in tar :
79 alldata.data[tar+sp+'%.8i'%seg]=gf(alldata.data[tar+sp+'%.8i'%seg],1)
80 i_null = np.where(alldata.data[tar+sp+'%.8i'%seg]==0.0)
81 alldata.data[tar+sp+'%.8i'%seg][i_null]=1.e-5
82
83#----compute-derived-fields------------------
84 print('COMPUTE->',seg)
85 for tar in out_tars:
86 if tar=='Vi' :
87 alldata.calc_divid('Ji','rhoi1',new_tar=tar,seg=seg)
88 if tar=='Ve' :
89 alldata.calc_divid('Je','rhoe0',new_tar=tar,seg=seg)
90 if 'T' in tar :
91 for sp in alldata.meta['species']:
92 alldata.calc_temp(tar.replace('T','P')+sp,new_tar=tar+sp,seg=seg)
93
94 print(alldata.data.keys())
95
96#----print-derived-fields-----------------------
97 print('PRINT->',seg)
98 for tar in out_tars:
99 if len(tar)<3 :
100 alldata.print_vtk_vect(tar+'_x',tar+'_y',tar+'_z',seg,data_address,alldata.meta['name']+'_'+tar+cut+'_'+str(seg),silent=True)
101 else :
102 for sp in alldata.meta['species']:
103 alldata.print_vtk_scal(tar+sp,seg,data_address,alldata.meta['name']+'_'+tar+sp+cut+'_'+str(seg),silent=True)
104
105#----del-3D-fields----------------------------
106 print('DELETE->',seg)
107 for key in alldata.data.keys():
108 del alldata.data[key]
109
110#----print-comp-time--------------------
111 print('COMP.TIME [seg%i] = %.3f s\n'%(seg,time.time()-t0))
