154def unpack(data_file,iy,iz,string):
155
156 # XY cut of the 3d array
157 if(iz==None):
158 # vectors
159 if(string=='Bx'):
160 return convert_vtk_vect(data_file,'B')[0][::fact,iy,::fact]
161 if(string=='By'):
162 return convert_vtk_vect(data_file,'B')[1][::fact,iy,::fact]
163 if(string=='Bz'):
164 return convert_vtk_vect(data_file,'B')[2][::fact,iy,::fact]
165 if(string=='Ex'):
166 return convert_vtk_vect(data_file,'E')[0][::fact,iy,::fact]
167 if(string=='Ey'):
168 return convert_vtk_vect(data_file,'E')[1][::fact,iy,::fact]
169 if(string=='Ez'):
170 return convert_vtk_vect(data_file,'E')[2][::fact,iy,::fact]
171 if(string=='Jxi'):
172 return convert_vtk_vect(data_file,'Ji')[0][::fact,iy,::fact]
173 if(string=='Jyi'):
174 return convert_vtk_vect(data_file,'Ji')[1][::fact,iy,::fact]
175 if(string=='Jzi'):

177 if(string=='Jxe'):
178 return convert_vtk_vect(data_file,'Je')[0][::fact,iy,::fact]
179 if(string=='Jye'):
180 return convert_vtk_vect(data_file,'Je')[1][::fact,iy,::fact]
181 if(string=='Jze'):
182 return convert_vtk_vect(data_file,'Je')[2][::fact,iy,::fact]
183 # scalars
184 scal = convert_vtk_scal(data_file)
185 return scal[::fact,iy,::fact]

188 if(iy==None):
189 # vectors
190 if(string=='Bx'):
191 return convert_vtk_vect(data_file,'B')[0][::fact,::fact,iz]
