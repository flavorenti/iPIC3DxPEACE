114 range_y = None,
115 range_z = None):
116 """
117 Finds points inside a range of coordinates
118 (equivalent to find_common(tar_pnts,find_box(range_x,...)) but much faster)
119
120 Parameters :
121 - tar_pnts [fibo.pnts] point set
122 - range_x [int,int]
123 - range_y [int,int]
124 - range_z [int,int]
125
126 Returns :
127 - ok_pnts [fibo.pnts] points inside range
128
129 """
130
131 nx,ny,nz = self.meta['nnn']
132
133 if range_x is None : range_x = [0,nx]
134 if range_y is None : range_y = [0,ny]
135 if range_z is None : range_z = [0,nz]
136
137 mask_pnt = np.zeros(self.meta['nnn'],dtype='bool')
138 mask_ran = np.zeros(self.meta['nnn'],dtype='bool')
139
140 # let's put the points into a binary mask
141 mask_pnt[tuple(self.get_pnts(tar_pnts).astype('int'))] = True
142
143 # and create another binary mask on the point range
144 ran_x,ran_y,ran_z = self.axis_data(range_x,range_y,range_z,mesh_like=True,wrap_like=True)
145 mask_ran[ran_x,ran_y,ran_z] = True
146
147 # then compare the masks
148 mask = np.logical_and(mask_pnt,mask_ran)
149
150 # and return the common points
151 return np.array( np.where(mask) )

156 def find_critical_squarely_2d(self,
157 tar_var,
158 cut_z=0,
159 aa=1,
160 bb=0,
161 thr=1e-1):
162 """
163 Finds zeros in 2d array by triangulation on neighbors
164 (x-y-periodic only)
165
166 Parameters :
167 - tar_var [fibo.data] target field
168 - cut_z = 0 [int] z coordinate of the cut considered
169 - aa = 1 [int>0] first parameter to determine neighbors' diamond
170 - bb = 0 [int>0] second parameter to determine neighbors' diamond
171 - thr=1e-1 [float>0] threshold value for maxima/minima detection
172
173 Returns :
174 - coord_Oa [fibo.pnts] maxima (O-points above)
175 - coord_Ob [fibo.pnts] minima (O-points below)
176 - coord_X [fibo.pnts] saddle (X-points)
177
178 """
179 nx,ny,nz = self.meta['nnn']
180 arr = np.tile(self.get_data(tar_var)[:,:,cut_z],(2,2))
181
182 #create eight (nx,ny) arrays with all differences between one value and its neighbours
183 neigh = np.zeros([8,nx,ny])
184
185 neigh[0,:,:] = (arr[aa:nx+aa, bb:ny+bb] - arr[0:nx, 0:ny])
186 neigh[2,:,:] = (arr[nx-bb:2*nx-bb, aa:ny+aa] - arr[0:nx, 0:ny])
187 neigh[4,:,:] = (arr[nx-aa:2*nx-aa, ny-bb:2*ny-bb] - arr[0:nx, 0:ny])
188 neigh[6,:,:] = (arr[bb:nx+bb, ny-aa:2*ny-aa] - arr[0:nx, 0:ny])
189
190 if aa > bb :
191 neigh[1,:,:] = (arr[ aa-bb:nx+aa-bb, aa+bb:ny+aa+bb] - arr[0:nx, 0:ny])
192 neigh[3,:,:] = (arr[ nx-aa-bb:2*nx-aa-bb, aa-bb:ny+aa-bb] - arr[0:nx, 0:ny])
193 neigh[5,:,:] = (arr[ nx-aa+bb:2*nx-aa+bb, ny-aa-bb:2*ny-aa-bb] - arr[0:nx, 0:ny])
194 neigh[7,:,:] = (arr[ aa+bb:nx+aa+bb, ny-aa+bb:2*ny-aa+bb] - arr[0:nx, 0:ny])
195 else :
196 neigh[1,:,:] = (arr[ nx+aa-bb:2*nx+aa-bb, aa+bb:ny+aa+bb] - arr[0:nx, 0:ny])
197 neigh[3,:,:] = (arr[ nx-aa-bb:2*nx-aa-bb, ny+aa-bb:2*ny+aa-bb] - arr[0:nx, 0:ny])
198 neigh[5,:,:] = (arr[-aa+bb:nx-aa+bb, ny-aa-bb:2*ny-aa-bb] - arr[0:nx, 0:ny])
199 neigh[7,:,:] = (arr[aa+bb:nx+aa+bb, -aa+bb:ny-aa+bb] - arr[0:nx, 0:ny])
200
201 #calculate Delta plus and minus
202 pos_neighs = np.maximum(neigh, np.zeros([8,nx,ny]))
203 Delta_plus = np.sum(pos_neighs,axis=0)
204 Delta_minus = np.sum(pos_neighs-neigh,axis=0)
205
206 #select positive neighbors and calculate sign changes
207 pos_sel = pos_neighs > 0.
208 sign_change = np.sum(np.diff(pos_sel,axis=0),axis=0) + 1
209 sign_change -= sign_change % 2
210
211 #calculate coordinates of Oa points (maxima), Ob points (minima) and X points (saddles)
212 coord_Oa = np.transpose(np.argwhere(np.logical_and(Delta_plus < thr*Delta_minus, sign_change == 0)))
213 coord_Ob = np.transpose(np.argwhere(np.logical_and(Delta_minus < thr*Delta_plus, sign_change == 0)))
214 coord_X = np.transpose(np.argwhere(sign_change > 2)) #np.logical_and(Delta_plus-Delta_minus < thr*(Delta_plus+Delta_minus), sign_change > 2)))
215
216 coord_Oa = (np.vstack([coord_Oa[0],coord_Oa[1],np.ones(np.shape(coord_Oa)[1]) * cut_z])).astype('int')
217 coord_Ob = (np.vstack([coord_Ob[0],coord_Ob[1],np.ones(np.shape(coord_Ob)[1]) * cut_z])).astype('int')
218 coord_X = (np.vstack([coord_X[0],coord_X[1],np.ones(np.shape(coord_X)[1]) * cut_z])).astype('int')
219
220 return coord_Oa, coord_Ob, coord_X

409 def find_middle_pnts(self,
410 tar_pnts,
411 bar_pnt = True,
412 med_pnt = True):
413 """
414 Finds relavant points in a given point-cluster
415
416 Parameters :
417 - tar_pnts [fibo.pnts] set of points
418 - bar_pnt [bool] do you want the barycentre?
419 - med_pnt [bool] do you want the medoid?
420
421 Returns :
422 - bar_coord [fibo.pnts] barycentre coordinates
423 - med_coord [fibo.pnts] medoid coordintes
424
425 """
426
427 nx,ny,nz = self.meta['nnn']
428 dx,dy,dz = self.meta['ddd']
429 pnts = self.get_pnts(tar_pnts)
430
431 # here I determine how many relevant periodic dimensions are there
432 dim_x = self.meta['ppp'][0] and self.meta['nnn'][0] != 1
433 dim_y = self.meta['ppp'][1] and self.meta['nnn'][1] != 1
434 dim_z = self.meta['ppp'][2] and self.meta['nnn'][2] != 1
435
436 # create the relevant-point coordinate arrays
437 med = np.zeros(3)
438 bar = np.zeros(3)
439
440 # let's find the medoid first
441 # it's easier given that we are in a usually-periodic environment
