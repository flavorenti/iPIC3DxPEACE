2# Code based on fibo library that (1) READS, (2) COMPUTE Anis,Gyro,mu=sqrt(Tper)/B
3# and (3) PRINT the new fields. Uses the derived two-dimensional
4# and three-dimensional output of an iPIC3D simu.
5# The new fields are saved in the same dir as the data, with the same dimensions.
6# Add global path to data dir in command line.
7# Only parameters to change are:
8#
9# inp_tars = [str] target variables you want to use (in order!)
10# out_tars = [str] name of the variables creates to print (only scalars!) must follow the order!
11# cuts = [str] cuts to use if the data are 2D (eg data1), if [''] data are 3D (eg data2)
12# cycle_min = [int] minimum cycle to read
13# cycle_max = [int] maximum cycle to read
14#
15# Extra rules for out_tars: Epar require Eper, Eprime require Eper, Tper require Tiso,
16# Tpar require Tper
17#
18# Job 12/2021
19#-------------------------------------------------------------------
20
21######################################################################################
22#======regulate=parameters=============================================================
23inp_tars = ['B','TXX','TXY','TXZ','TYY','TYZ','TZZ','Tpar','Tper']
24out_tars = ['Anis','Gyro','mu']
25cuts = ['eq','dp']
26cycle_min = 0
27cycle_max = 1000000
28#=======================================================================================
29########################################################################################
30
31########################################################################################
32####################### DO NOT TOUCH BELOW! ############################################

34#----import-modules---------------
35import fibo as fb
36import time
37import sys
38import numpy as np
39
40#----path-to-run-------------------
41data_address = sys.argv[1]
42
43#----create-your-objects-----------
44fibo_name = 'ipic3d'
45alldata = fb.fibo(fibo_name)
46from_VTK = fb.from_VTK(data_address)
47
48#----loop-on-types-of-data--------
49for cut in cuts:
50
51 #----load-metadata------------------
52 from_VTK.get_meta(silent=False)
53 if cut == 'eq':
54 from_VTK.meta['nz']=1
55 from_VTK.meta['nnn']=(from_VTK.meta['nx'],from_VTK.meta['ny'],1)
56 if cut == 'dp':
57 from_VTK.meta['ny']=1
58 from_VTK.meta['nnn']=(from_VTK.meta['nx'],1,from_VTK.meta['nz'])
59 alldata.meta = from_VTK.meta
60
61 #----load-everything-in-code-units-------------------
62 for seg in alldata.meta['segcycles']:
63 t0 = time.time()
64 if cycle_min<=seg<=cycle_max :
65 print('LOAD->',seg)
66 for tar in inp_tars:
67 if len(tar)<3 :
68 from_VTK.get_vect(alldata.meta['name']+'_'+tar+cut+'_'+str(seg),seg,fibo_obj=alldata,tar_var=tar,silent=True)
69 else :
70 for sp in alldata.meta['species']:
71 from_VTK.get_scal(alldata.meta['name']+'_'+tar+sp+cut+'_'+str(seg),seg,fibo_obj=alldata,tar_var=tar+sp,silent=True)
72
73 print(alldata.data.keys())
74
75#----compute-derived-fields------------------
76 print('COMPUTE->',seg)
77 for sp in alldata.meta['species']:
78 I1 = alldata.data['TXX'+sp+'%.8i'%int(seg)]+alldata.data['TYY'+sp+'%.8i'%int(seg)]+alldata.data['TZZ'+sp+'%.8i'%int(seg)]
79 I2 = alldata.calc_scalr('TXX'+sp,'TYY'+sp,'TXX'+sp,'TZZ'+sp,'TYY'+sp,'TZZ'+sp,seg=seg,fill=False)
80 I2 -= alldata.calc_scalr('TXY'+sp,'TXY'+sp,'TXZ'+sp,'TXZ'+sp,'TYZ'+sp,'TYZ'+sp,seg=seg,fill=False)
81 modB2 = alldata.calc_scalr('B_x','B_x','B_y','B_y','B_z','B_z',seg=seg,fill=False)
82 for tar in out_tars:
83 if tar=='Anis' :
84 alldata.data[tar+sp+'%.8i'%int(seg)] = 2.*alldata.data['Tpar'+sp+'%.8i'%int(seg)]/alldata.data['Tper'+sp+'%.8i'%int(seg)]
85 if tar=='Gyro' :
86 alldata.data[tar+sp+'%.8i'%int(seg)] = 1. - (4.*I2/(I1-alldata.data['Tpar'+sp+'%.8i'%int(seg)])/(I1+3.*alldata.data['Tpar'+sp+'%.8i'%int(seg)]))
87 if tar=='mu' :
88 alldata.data[tar+sp+'%.8i'%int(seg)] = np.sqrt(alldata.data['Tper'+sp+'%.8i'%int(seg)]/modB2)
89
90 print(alldata.data.keys())
91
92
93#----print-derived-fields-----------------------
94 print('PRINT->',seg)
95 for tar in out_tars:
96 for sp in alldata.meta['species']:
97 alldata.print_vtk_scal(tar+sp,seg,data_address,alldata.meta['name']+'_'+tar+sp+cut+'_'+str(seg),silent=True)
98
99#----del-3D-fields----------------------------
100 print('DELETE->',seg)
101 del I1,I2
102 for key in alldata.data.keys():
103 del alldata.data[key]
104
105#----print-comp-time--------------------
106 print('COMP.TIME [seg%i] = %.3f s\n'%(seg,time.time()-t0))
