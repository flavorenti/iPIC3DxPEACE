12def load_cusp(input_file_cusp,proj=None):
13
14 # load txt file with values cusp thresold
15 phi_c,theta_c,comment = np.loadtxt(input_file_cusp,unpack=True,dtype='str')
16 lot_cusp = np.array(phi_c,dtype='float')*24./(2*np.pi)
17 lat_cusp = 90.-(np.array(theta_c,dtype='float')*180./np.pi)
18 x_cusp=[]
19 y_cusp=[]
20 for i in range(0,len(phi_c)):
21 if lot_cusp[i] not in x_cusp:
22 if len(lat_cusp[(phi_c==phi_c[i])*(lat_cusp>0)])>0:
23 x_cusp.append(lot_cusp[i])
24 y_cusp.append(np.min(lat_cusp[(phi_c==phi_c[i])*(lat_cusp>0)]))
25 if len(lat_cusp[(phi_c==phi_c[i])*(lat_cusp<0)])>0:
26 x_cusp.append(lot_cusp[i])
27 y_cusp.append(np.max(lat_cusp[(phi_c==phi_c[i])*(lat_cusp<0)]))
28 x_cusp = np.array(x_cusp)
29 y_cusp = np.array(y_cusp)
30 x_cuspN = x_cusp[np.where(y_cusp>0)]
31 y_cuspN = y_cusp[np.where(y_cusp>0)]
32 x_cuspS = x_cusp[np.where(y_cusp<0)]
33 y_cuspS = y_cusp[np.where(y_cusp<0)]
34
35 # convert to good units for mollweide proj.
36 if proj is not None:
37 x_cuspN = (x_cuspN-12.)/12.*np.pi
38 x_cuspS = (x_cuspS-12.)/12.*np.pi
39 y_cuspN = y_cuspN/90.*(np.pi/2.)
40 y_cuspS = y_cuspS/90.*(np.pi/2.)
41
42 print('CUSP LOADED -->'+input_file_cusp)
43
44 return [[x_cuspN,y_cuspN],[x_cuspS,y_cuspS]]
