53/*
54 * drf.c: Compute the complete or incomplete elliptic integral of the
55 * first kind.
56 *
57 * Description:
58 *
59 * For x,y and z non-negative and at most one of them zero, drf(x,y,z)
60 * = integral from zero to infinity of
61 *
62 *
63 * -1/2 -1/2 -1/2
64 * (1/2)(t+x) (t+y) (t+z) dt.
65 *
66 * If x, y or z is zero, the integral is complete.
67 *
68 * *piErr returns non-zero if any arguments are invalid.
69 *
70 * Credits:
71 *
72 * This function is adapted by E. Dennison from FORTRAN code written by:
73 *
74 * Carlson, B.C.
75 * Notis, E.M
76 * Pexton, R.L.
77 *
78 */
79
80double drf(double x, double y, double z, int* piErr)
81{
82int iErr=0;
83double mu,
84 xn,yn,zn,
85 xndev,yndev,zndev,
86 xnroot,ynroot,znroot,
87 lambda,
88 epsilon,
89 e2,e3,
90 result,
91 s;
92
93 const double c1 = 1.0/24.0;
94 const double c2 = 3.0/44.0;
95 const double c3 = 1.0/14.0;
96 const double errtol = pow(DBL_EPSILON*4.0,1.0/6.0);
97 const double lolim = 5.0*DBL_MIN;
98 const double hilim = DBL_MAX/5.0;
99
100 if (piErr)
101 {
102 if (MIN3(x,y,z)<0.0)
103 {
104 iErr = 1;
105 }
106 else if (MIN3(x+y,x+z,y+z)<lolim)
107 {
108 iErr = 2;
109 }
110 else if (MAX3(x,y,z)>hilim)
111 {
112 iErr = 3;
113 }
114 }
115 if (iErr)
116 {
117 if (piErr)
118 {
119 *piErr = iErr;
120 }
121 result = 0.0;
122 }
123
124else
125 {
126 xn = x;
127 yn = y;
128 zn = z;
129
130 while (1)
131 {
132 mu = (xn+yn+zn)/3.0;
133 xndev = 2.0-(mu+xn)/mu;
134 yndev = 2.0-(mu+yn)/mu;
135 zndev = 2.0-(mu+zn)/mu;
136 epsilon = MAX3(fabs(xndev),fabs(yndev),fabs(zndev));
137 if (epsilon < errtol) break;
138 xnroot = sqrt(xn);
139 ynroot = sqrt(yn);
140 znroot = sqrt(zn);
141 lambda = xnroot*(ynroot+znroot) +ynroot*znroot;
142 xn = (xn+lambda)*0.25;
143 yn = (yn+lambda)*0.25;
144 zn = (zn+lambda)*0.25;
145 }
146 e2 = xndev*yndev - pow(zndev,2);
147 e3 = xndev*yndev*zndev;
148 s = 1.0 + (c1*e2-0.1-c2*e3)*e2 + c3*e3;
149
150 if (piErr)
151 {
152 *piErr = 0;
153 }
154 result = s/sqrt(mu);
155 }
156 return result;
157}
158/* END drf() */

177/*
178 * drd.c: Compute the complete or incomplete elliptic integral of the
179 * second kind.
180 *
181 * Description:
182 *
183 * For x and y non-negative, x+y and z positive, drf(x,y,z) = integral
184 * from zero to infinity of
185 *
186 *
187 * -1/2 -1/2 -3/2
188 * (3/2)(t+x) (t+y) (t+z) dt.
189 *
190 * If x or y is zero, the integral is complete.
191 *
192 * *piErr returns non-zero if any arguments are invalid.
193 *
194 * Credits:
195 *
196 * This function is adapted by E. Dennison from FORTRAN code written by:
197 *
198 * Carlson, B.C.
199 * Notis, E.M
200 * Pexton, R.L.
201 *
202 */
203
204double drd(double x, double y, double z, int* piErr)
205{
206 int iErr=0;
207 double mu,
208 xn,yn,zn,
209 xndev,yndev,zndev,
210 xnroot,ynroot,znroot,
211 lambda,
212 epsilon,
213 ea,eb,ec,ed,ef,
214 sigma,
215 power4,
216 result,
217 s1,s2;
218
219 const double c1 = 3.0/14.0;
220 const double c2 = 1.0/6.0;
221 const double c3 = 9.0/22.0;
222 const double c4 = 3.0/26.0;
223 const double errtol = pow(DBL_EPSILON/3.0,1.0/6.0);
224 double uplim;
225 const double lolim = 2.0/pow(DBL_MAX,2.0/3.0);
226 double tuplim = pow(DBL_MIN,1.0/3.0);
227 tuplim = pow(0.1*errtol,1.0/3.0)/tuplim;
228 uplim = pow(tuplim,2.0);
229
230 if (piErr)
231 {
232 if (MIN(x,y)<0.0)
233 {
234 iErr = 1;
235 }
236 else if (MAX3(x,y,z)>uplim)
237 {
238 iErr = 2;
239 }
240 else if (MIN(x+y,z)<lolim)
241 {
242 iErr = 3;
243 }
244 }
245 if (iErr)
246 {
247 if (piErr)
248 {
249 *piErr = iErr;
250 }
251 result = 0.0;
252 }
253
254else
255 {
256 xn = x;
257 yn = y;
258 zn = z;
259 sigma = 0.0;
260 power4 = 1.0;
261 while (1)
262 {
263 mu = (xn+yn+3.0*zn)*0.2;
264 xndev = (mu-xn)/mu;
265 yndev = (mu-yn)/mu;
266 zndev = (mu-zn)/mu;
267 epsilon = MAX3(fabs(xndev),fabs(yndev),fabs(zndev));
268 if (epsilon < errtol) break;
269 xnroot = sqrt(xn);
270 ynroot = sqrt(yn);
271 znroot = sqrt(zn);
272 lambda = xnroot*(ynroot+znroot) +ynroot*znroot;
273 sigma = sigma+power4/(znroot*(zn+lambda));
274 power4 = power4*0.25;
275 xn = (xn+lambda)*0.25;
276 yn = (yn+lambda)*0.25;
277 zn = (zn+lambda)*0.25;
278 }
279 ea = xndev*yndev;
280 eb = zndev*zndev;
281 ec = ea-eb;
282 ed = ea-6.0*eb;
283 ef = ed+ec+ec;
284 s1 = ed*(-c1+0.25*c3*ed-1.5*c4*zndev*ef);
285 s2 = zndev*(c2*ef+zndev*(-c3*ec+zndev*c4*ea));
286 if (piErr)
287 {
288 *piErr = 0;
289 }
290 result = 3.0*sigma+power4*(1.0+s1+s2)/(mu*sqrt(mu));
291 }
292 return result;
293}
