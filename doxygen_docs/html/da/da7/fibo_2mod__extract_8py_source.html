20 def extract_line(self,
21 tar_data,
22 line_pnts,
23 line_dims,
24 center,
25 versor):
26 """
27 Extracts array values along a specified line
28
29 Parameters :
30 - tar_data [fibo.data] target variable
31 - line_pnts [int] length (in points) of the line you ask to extract
32 - line_dims [float] dimension of the line you ask to extract
33 - center [float,float,float] coordinates of the first and last point
34 - versor [float,float,float] projections of the versor over which the line is built
35
36 Returns :
37 - new_data [fibo.data] new data
38 - new_meta [fibo.meta] new metadata
39
40 """
41
42 nx,ny,nz = self.meta['nnn']
43 dx,dy,dz = self.meta['ddd']
44
45 # here I determine how many relevant periodic dimensions there are
46 dim_x = self.meta['ppp'][0] and self.meta['nnn'][0] != 1
47 dim_y = self.meta['ppp'][1] and self.meta['nnn'][1] != 1
48 dim_z = self.meta['ppp'][2] and self.meta['nnn'][2] != 1
49
50 # then double the box in every relevant periodic dimension
51 arr = self.get_data(tar_data)
52
53 if dim_x : arr = np.tile(arr,(2,1,1))
54 if dim_y : arr = np.tile(arr,(1,2,1))
55 if dim_z : arr = np.tile(arr,(1,1,2))
56
57 # now let's move the center into the central part of this extended array
58 if dim_x : center[0] += (1 - 2*center[0]//nx)*nx
59 if dim_y : center[1] += (1 - 2*center[1]//ny)*ny
60 if dim_z : center[2] += (1 - 2*center[2]//nz)*nz
61
62 # re-scale the in-plane versors by the dimensions
63 vec = np.array(versor) * line_dims
64
65 # then, add and subtract the in-plane vectors from the plane center coordinates
66 rx = np.array([center[0]]*2)
67 rx[0] += 0.5* ( -vec[0] ) / dx
68 rx[1] += 0.5* ( +vec[0] ) / dx
69 ry = np.array([center[1]]*2)
70 ry[0] += 0.5* ( -vec[1] ) / dy
71 ry[1] += 0.5* ( +vec[1] ) / dy
72 rz = np.array([center[2]]*2)
73 rz[0] += 0.5* ( -vec[2] ) / dz
74 rz[1] += 0.5* ( +vec[2] ) / dz
75
76 # so that you can finally calculate the new data
77 lin_x = np.linspace(rx[0],rx[1],line_pnts,endpoint=True)
78 lin_y = np.linspace(ry[0],ry[1],line_pnts,endpoint=True)
79 lin_z = np.linspace(rz[0],rz[1],line_pnts,endpoint=True)
80
81 new_data = ndm.map_coordinates(arr, np.vstack((lin_x,lin_y,lin_z)))
82 new_data = new_data.reshape((line_pnts,1,1))
83
84 # and the new meta data
85 new_meta = self.meta.copy()
86 new_meta['space_dim'] = '1D'
87
88 new_meta['nx'] = line_pnts
89 new_meta['ny'] = 1
90 new_meta['nz'] = 1
91
92 new_meta['xl'] = line_dims
93 new_meta['yl'] = 2*np.pi
94 new_meta['zl'] = 2*np.pi
95
96 new_meta['dx'] = new_meta['xl'] / new_meta['nx']
97 new_meta['dy'] = new_meta['yl'] / new_meta['ny']
98 new_meta['dz'] = new_meta['zl'] / new_meta['nz']
99
100 new_meta['nnn'] = (new_meta['nx'],new_meta['ny'],new_meta['nz'])
101 new_meta['lll'] = (new_meta['xl'],new_meta['yl'],new_meta['zl'])
102 new_meta['ddd'] = (new_meta['dx'],new_meta['dy'],new_meta['dz'])
103
104 del(new_meta['np_col'])
105 del(new_meta['np_pla'])
106 del(new_meta['np_row'])
107
108 return new_data, new_meta

124 - grid_dims [float,float] dimensions of the plane you ask to extract
125 - center [float,float,float] coordinates of the central point
126 - versor_one [float,float,float] projections of the first in-plane versor
127 - versor_two [float,float,float] projections of the second in-plane versor

129 Returns :
130 - new_data [fibo.data] new data
131 - new_meta [fibo.meta] new metadata
132

138 # here I determine how many relevant periodic dimensions there are
139 dim_x = self.meta['ppp'][0] and self.meta['nnn'][0] != 1
140 dim_y = self.meta['ppp'][1] and self.meta['nnn'][1] != 1
141 dim_z = self.meta['ppp'][2] and self.meta['nnn'][2] != 1
142
143 # then double the box in every relevant periodic dimension
144 arr = self.get_data(tar_data)
145
146 if dim_x : arr = np.tile(arr,(2,1,1))
147 if dim_y : arr = np.tile(arr,(1,2,1))

150 # now let's move the center into the central part of this extended array

153 if dim_z : center[2] += (1 - 2*center[2]//nz)*nz

159 # then, add and subtract the in-plane vectors from the plane center coordinates

162 rx[1] += 0.5* ( +vec_one[0] -vec_two[0] ) / dx
163 rx[2] += 0.5* ( -vec_one[0] +vec_two[0] ) / dx

166 ry[1] += 0.5* ( +vec_one[1] -vec_two[1] ) / dy
167 ry[2] += 0.5* ( -vec_one[1] +vec_two[1] ) / dy
168 rz = np.array([float(center[2])]*3)
169 rz[0] += 0.5* ( -vec_one[2] -vec_two[2] ) / dz
170 rz[1] += 0.5* ( +vec_one[2] -vec_two[2] ) / dz
171 rz[2] += 0.5* ( -vec_one[2] +vec_two[2] ) / dz
172
173 # so that you can finally calculate the new data

177
178 sec_x = np.ndarray.flatten(sec_x+sec_xx) - rx[0]
179 sec_y = np.ndarray.flatten(sec_y+sec_yy) - ry[0]
180 sec_z = np.ndarray.flatten(sec_z+sec_zz) - rz[0]
181
182 new_data = np.transpose(np.reshape(ndm.map_coordinates(arr, np.vstack((sec_x,sec_y,sec_z))),(grid_pnts[1],grid_pnts[0])))
183 new_data = new_data.reshape((grid_pnts[0],grid_pnts[1],1))
184
185 # and the new meta data
186 new_meta = self.meta.copy()
187 new_meta['space_dim'] = '2D'
188
189 new_meta['nx'] = grid_pnts[0]
190 new_meta['ny'] = grid_pnts[1]
191 new_meta['nz'] = 1
192
193 new_meta['xl'] = grid_dims[0]
194 new_meta['yl'] = grid_dims[1]
195 new_meta['zl'] = 2*np.pi
196
197 new_meta['dx'] = new_meta['xl'] / new_meta['nx']
198 new_meta['dy'] = new_meta['yl'] / new_meta['ny']
199 new_meta['dz'] = new_meta['zl'] / new_meta['nz']
200
201 new_meta['nnn'] = (new_meta['nx'],new_meta['ny'],new_meta['nz'])
202 new_meta['lll'] = (new_meta['xl'],new_meta['yl'],new_meta['zl'])
203 new_meta['ddd'] = (new_meta['dx'],new_meta['dy'],new_meta['dz'])
204
205 del(new_meta['np_col'])
206 del(new_meta['np_pla'])
207 del(new_meta['np_row'])

294 def extract_every(self,
295 tar_data,
296 every_x,
297 every_y,
298 every_z):
299 """
300 Lowers resolution of array, generating a new array by taking one point every ...
301
302 Parameters :
303 - tar_data [fibo.data] target variable
304 - every_x [int] points to be kept in x (one every ratio_x)
305 - every_y [int] points to be kept in y (one every ratio_y)
306 - every_z [int] points to be kept in z (one every ratio_z)
307
308 Returns :
309 - new_data [np.ndarray(nx//ratio_x,ny//ratio_y,nz//ratio_z)] new variable
310 - new_meta [dict] new metadata
311
312 """
313
314 #determine the shape of the new array
315 arr = self.get_data(tar_data)
316 nx,ny,nz = self.meta['nnn']
317
318 nnx = int((nx+every_x-1)/every_x)
319 nny = int((ny+every_y-1)/every_y)
320 nnz = int((nz+every_z-1)/every_z)
321
322 #create and fill in the new array
323 new_data = np.zeros([nnx,nny,nnz], dtype=float)
324
325 for iz in range(0,nz,every_z):
326 for iy in range(0,ny,every_y):
327 for ix in range(0,nx,every_x):
328 new_data[ix//every_x,iy//every_y,iz//every_z] = arr[ix,iy,iz]
329
330 #print('done with the new array!')
331
332 #create new metadata for the object
333 new_meta = {}
334 new_meta['nnn'] = (nnx, nny, nnz)
335 new_meta['lll'] = self.meta['lll']
336 new_meta['ddd'] = tuple( np.array(self.meta['ddd']) * np.array([every_x,every_y,every_z]) )
337 new_meta['ppp'] = (np.array([nx,ny,nz]) % np.array([every_x,every_y,every_z])).astype('bool')
338 new_meta['ppp'] = tuple( np.logical_not(new_meta['ppp']) )
339
340 new_meta['nx'],new_meta['ny'],new_meta['nz'] = new_meta['nnn']
341 new_meta['xl'],new_meta['yl'],new_meta['zl'] = new_meta['lll']
342 new_meta['dx'],new_meta['dy'],new_meta['dz'] = new_meta['ddd']
343
344 return new_data, new_meta

347 def extract_interpol_fourier(self,
348 tar_data,
349 ratio_x,
350 ratio_y,
351 ratio_z):
352 """
353 Increases resolution of array by perfect Fourier-interpolation
354 (obviously, it acts only over periodic dimensions)
355
356 Parameters :
357 - tar_data [fibo.data] target variable
358 - ratio_x [int] ratio of point increment in x
359 - ratio_y [int] ratio of point increment in y
360 - ratio_z [int] ratio of point increment in z
361
362 Returns :
363 - new_data [np.ndarray(nx*ratio_x,ny*ratio_y,nz*ratio_z)] new variable
364 - new_meta [dict] new metadata
365
366 """
367
368 nx,ny,nz = self.meta['nnn']
369 nnx,nny,nnz = self.meta['nnn']
370
371 # here I determine how many relevant periodic dimensions are there
372 dim_x = self.meta['ppp'][0] and self.meta['nnn'][0] != 1
373 dim_y = self.meta['ppp'][1] and self.meta['nnn'][1] != 1
374 dim_z = self.meta['ppp'][2] and self.meta['nnn'][2] != 1
375
376 # let's perform the Fourier transform
377 arr = np.fft.fftn( self.get_data(tar_data) )
378
379 # here I widen the Fourier-transformed array, adding zeros
380 if dim_x :
381 zer = int(nx*(ratio_x-1.))
382 arr = np.concatenate((arr[:nx//2,:,:], np.zeros((zer,nny,nnz)),arr[nx//2:,:,:]), axis=0)
383 nnx = nx + zer
384 if dim_y :
385 zer = int(ny*(ratio_y-1.))
386 arr = np.concatenate((arr[:,:ny//2,:], np.zeros((nnx,zer,nnz)),arr[:,ny//2:,:]), axis=1)
387 nny = ny + zer
388 if dim_z :
389 zer = int(nz*(ratio_z-1.))
390 arr = np.concatenate((arr[:,:,:nz//2], np.zeros((nnx,nny,zer)),arr[:,:,nz//2:]), axis=2)
391 nnz = nz + zer
392
393 # now find the actual ratios
394 ratio_x = nnx / nx
395 ratio_y = nny / ny
396 ratio_z = nnz / nz
397
398 # now let's anti-transform the array:
399 new_data = np.fft.ifftn( arr ).real
400 if dim_x : new_data *= ratio_x
401 if dim_y : new_data *= ratio_y
402 if dim_z : new_data *= ratio_z
403
404 # and let's provide the meta-data for this new object
405 new_meta = {}
406 new_meta['nnn'] = (nnx, nny, nnz)
407 new_meta['lll'] = self.meta['lll']
408 new_meta['ddd'] = tuple( np.array(self.meta['ddd']) / np.array([ratio_x,ratio_y,ratio_z]) )
409 new_meta['ppp'] = np.copy(self.meta['ppp'])
410
411 new_meta['nx'],new_meta['ny'],new_meta['nz'] = new_meta['nnn']
412 new_meta['xl'],new_meta['yl'],new_meta['zl'] = new_meta['lll']
413 new_meta['dx'],new_meta['dy'],new_meta['dz'] = new_meta['ddd']
414
415 return new_data, new_meta
