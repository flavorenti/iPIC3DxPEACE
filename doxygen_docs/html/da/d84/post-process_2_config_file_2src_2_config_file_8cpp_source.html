62std::istream & operator>>(std::istream & is, ConfigFile & cf) {
63 // Load a ConfigFile from is
64 // Read in keys and values, keeping internal whitespace
65 typedef string::size_type pos;
66 const string & delim = cf.myDelimiter; // separator
67 const string & comm = cf.myComment; // comment
68 const string & sentry = cf.mySentry; // end of file sentry
69 const pos skip = delim.length(); // length of separator
70
71 string nextline = ""; // might need to read ahead to see where value ends
72
73 while (is || nextline.length() > 0) {
74 // Read an entire line at a time
75 string line;
76 if (nextline.length() > 0) {
77 line = nextline; // we read ahead; use it now
78 nextline = "";
79 }
80 else {
81 std::getline(is, line);
82 }
83
84 // Ignore comments
85 line = line.substr(0, line.find(comm));
86
87 // Check for end of file sentry
88 if (sentry != "" && line.find(sentry) != string::npos)
89 return is;
90
91 // Parse the line if it contains a delimiter
92 pos delimPos = line.find(delim);
93 if (delimPos < string::npos) {
94 // Extract the key
95 string key = line.substr(0, delimPos);
96 line.replace(0, delimPos + skip, "");
97
98 // See if value continues on the next line
99 // Stop at blank line, next line with a key, end of stream,
100 // or end of file sentry
101 bool terminate = false;
102 while (!terminate && is) {
103 std::getline(is, nextline);
104 terminate = true;
105
106 string nlcopy = nextline;
107 ConfigFile::trim(nlcopy);
108 if (nlcopy == "")
109 continue;
110
111 nextline = nextline.substr(0, nextline.find(comm));
112 if (nextline.find(delim) != string::npos)
113 continue;
114 if (sentry != "" && nextline.find(sentry) != string::npos)
115 continue;
116
117 nlcopy = nextline;
118 ConfigFile::trim(nlcopy);
119 if (nlcopy != "")
120 line += "\n";
121 line += nextline;
122 terminate = false;
123 }
124
125 // Store key and value
126 ConfigFile::trim(key);
127 ConfigFile::trim(line);
128 cf.myContents[key] = line; // overwrites if key is repeated
129 }
130 }
131
132 return is;
133}
