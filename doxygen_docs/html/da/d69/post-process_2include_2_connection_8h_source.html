26// The combination of group (comm), tag, and neighbor
27// should be unique for each connection.
28//
29// In a Cartesian topology, the challenges to this uniqueness
30// occur for periodic boundary conditions when there is a
31// dimension only 1 or 2 processes thick.
32//
33// In the 1-process-thick case, we can forgo MPI communication in
34// that direction altogether. But in the 2-process-thick case,
35// we need to make a distinction between upward and downward
36// channels of communication. Distinguishing upward and downward
37// directions is sufficient for any topology based on convex MPI
38// subdomains, since in this case two subdomains can share at
39// most two faces. Any mechanism for distinguishing upward and
40// downward directions for a pair of subdomains is sufficient;
41// one way is to compare the positions of the centroids, first in
42// the x direction, then in the y direction, and then in the z
43// direction. One might use tag=1 for downward communication and
44// tag=2 for upward communication, and all ways use particle communicator for the
45// group.
46//
47namespace Direction
48{
49 enum Enum
50 {
51 DEFAULT = 0,
52 PARTICLE_DN, // downward communication of particles
53 PARTICLE_UP, // upward communication of particles
54 XDN,
55 XUP,
56 YDN,
57 YUP,
58 ZDN,
59 ZUP
60 };
61}
62class Connection
63{
64 private: // data
65 // In MPI a message envelope includes the following
66 // information plus the rank of this process.
67 int _rank; // rank within group of neighbors we're connecting to
68 int _tag; // tag to attach to messages
69 MPI_Comm _comm; // communicator group
70 public: // init
71 // construct a connection that creates self-communication
72 // in place of null connection
73 static Connection null2self(int rank_, int tag_, int self_tag, MPI_Comm comm_)
74 {
75 Connection c(rank_,tag_,comm_);
76 if(c._rank==MPI_PROC_NULL)
77 {
78 c._rank = MPIdata::get_rank();
79 c._tag = self_tag;
80 }
81 return c;
82 }
83 Connection():
84 _rank(0),
85 _tag(0),
86 _comm(MPI_COMM_NULL)
87 {}
88 Connection(int rank_, int tag_, MPI_Comm comm_ = MPI_COMM_NULL)
89 {
90 init(rank_,tag_,comm_);
91 }
92 private:
93 void init(int rank_, int tag_, MPI_Comm comm_ = MPI_COMM_NULL)
94 {
95 _rank = rank_;
96 _tag = tag_;
97 _comm = comm_;
98 }
99 public: // accessors
100 int rank()const{return _rank;}
101 int tag()const{return _tag;}
102 MPI_Comm comm()const{return _comm;}
103 static const char* tag_name(int tag)
104 {
105 switch(tag)
106 {
107 default: unsupported_value_error(tag);
108 case Direction::XDN: return "XDN";
109 case Direction::XUP: return "XUP";
110 case Direction::YDN: return "YDN";
111 case Direction::YUP: return "YUP";
112 case Direction::ZDN: return "ZDN";
113 case Direction::ZUP: return "ZUP";
114 }
115 }
116 const char* tag_name()const
117 {
118 return tag_name(_tag);
119 }
120};
