21 def comp_distances(self,
22 tar_pnts,
23 all_comp):
24 """
25 Find distance matrix for a set of points
26
27 Parameters:
28 - tar_pnts [fibo.pnts] original points
29 - all_comp [bool] restitute all components of distances?
30
31 Returns :
32 - dist_mat [float array] distance matrix, with or without components specified
33
34 """
35
36 nx,ny,nz = self.meta['nnn']
37 dx,dy,dz = self.meta['ddd']
38 pnts = self.get_pnts(tar_pnts)
39
40 # here I determine how many relevant periodic dimensions are there
41 dim_x = self.meta['ppp'][0] and self.meta['nnn'][0] != 1
42 dim_y = self.meta['ppp'][1] and self.meta['nnn'][1] != 1
43 dim_z = self.meta['ppp'][2] and self.meta['nnn'][2] != 1
44
45 dist_mat_x = np.tile(pnts[0,:],(pnts.shape[1],1))
46 dist_mat_x -= np.transpose(dist_mat_x)
47 if dim_x :
48 dist_mat_x[ dist_mat_x > nx/2 ] -= nx
49 dist_mat_x[ dist_mat_x < -nx/2 ] += nx
50
51 dist_mat_y = np.tile(pnts[1,:],(pnts.shape[1],1))
52 dist_mat_y -= np.transpose(dist_mat_y)
53 if dim_y :
54 dist_mat_y[ dist_mat_y > ny/2 ] -= ny
55 dist_mat_y[ dist_mat_y < -ny/2 ] += ny
56
57 dist_mat_z = np.tile(pnts[2,:],(pnts.shape[1],1))
58 dist_mat_z -= np.transpose(dist_mat_z)
59 if dim_z :
60 dist_mat_z[ dist_mat_z > nz/2 ] -= nz
61 dist_mat_z[ dist_mat_z < -nz/2 ] += nz
62 #anti[2, dist_mat_z > nz/2-1 ] = True
63
64 if all_comp:
65 dist_mat = [dx*dist_mat_x, dy*dist_mat_y, dz*dist_mat_z]
66 else:
67 dist_mat = np.sqrt( np.square( dx*dist_mat_x ) + np.square( dy*dist_mat_y ) + np.square( dz*dist_mat_z ) )
68
69 return dist_mat

99 for k in range(pnts.shape[1]) :

122 def comp_spot_number(self,
123 tar_spots,
124 bin_struc = np.ones((3,3,3))):
125 """
126 Finds the number of connected spots
127
128 Parameters :
129 - tar_spots [fibo_data] binary mask individuating the spots
130 - bin_struc = np.ones((3,3,3)) [3x3x3 binary matrix] binary structure for blurring
131
132 Returns :
133 - spot_num [int>0] connected spots in your environment
134
135 """
136
137 nx,ny,nz = self.meta['nnn']
138 arr = self.get_data(tar_spots)
139
140 # here I determine how many relevant periodic dimensions are there
141 dim_x = self.meta['ppp'][0] and self.meta['nnn'][0] != 1
142 dim_y = self.meta['ppp'][1] and self.meta['nnn'][1] != 1
143 dim_z = self.meta['ppp'][2] and self.meta['nnn'][2] != 1
144
145 # let's double the array over every relevant periodic dimension
146 if dim_x : arr = np.concatenate( [arr[-1:,:,:],arr,arr[:1,:,:]] ,axis=0 )
147 if dim_y : arr = np.concatenate( [arr[:,-1:,:],arr,arr[:,:1,:]] ,axis=1 )
148 if dim_z : arr = np.concatenate( [arr[:,:,-1:],arr,arr[:,:,:1]] ,axis=2 )
149
150 # individuate the spot mask (this is the core of all)
151 spot_mask, spot_number = ndm.measurements.label(arr,structure=bin_struc)
152 # and fold the boundaries of the spot mask so that it stacks over the original array
153 # then you can check the stacked edges for label-correspondences
154 # and merge identical labels
155 # at the same time, look for the tiling that allows for the complete view of each spot
156
157 relabel = np.arange(1,spot_number+1)
158
159 def corr_check(relabel,edg,check):
160 # if check then perform re-labeling on the edge before checking correlations
161 if check:
162 for k in range(len(relabel)) :
163 kk = relabel[k]
164 if kk != k+1 :
165 edg[edg == k+1] = kk
166 # candidates for new labels
167 edg_labels = np.min(edg,axis=0)
168 # for every value that the label has in edg (except zero)
169 for k in np.flip( np.unique(edg)[1:] ) :
170 pos_k = np.where(edg == k)
171 kk = np.min( edg_labels[pos_k[1:]] )
172 # and - if different - substitute the latter to the former, everywhere
173 if k != kk :
174 edg[pos_k] = kk
175 edg_labels = np.min(edg,axis=0)
176 loc_k = np.where(relabel == k)
177 relabel[loc_k] = kk
178 #print('changed mask '+str(k)+' to '+str(kk))
179
180 if dim_x: corr_check(relabel,np.array([spot_mask[:2,:,:],spot_mask[-2:,:,:]]),False)
181 if dim_y: corr_check(relabel,np.array([spot_mask[:,:2,:],spot_mask[:,-2:,:]]),dim_x)
182 if dim_z: corr_check(relabel,np.array([spot_mask[:,:,:2],spot_mask[:,:,-2:]]),dim_x|dim_y)
183
184 # we're at the end: just re-name the labels now!
185 spot_number = len(np.unique(relabel))
186
187 return spot_number
