84int c_Solver::Init(int argc, char **argv) {
85 #if defined(__MIC__)
86 assert_eq(DVECWIDTH,8);
87 #endif
88 // get MPI data
89 //
90 // c_Solver is not a singleton, so the following line was pulled out.
91 //MPIdata::init(&argc, &argv);
92 //
93 // initialized MPI environment
94 // nprocs = number of processors
95 // myrank = rank of tha process*/
96 Parameters::init_parameters();
97 //mpi = &MPIdata::instance();
98 nprocs = MPIdata::get_nprocs();
99 myrank = MPIdata::get_rank();
100
101 col = new Collective(argc, argv); // Every proc loads the parameters of simulation from class Collective
102 restart_cycle = col->getRestartOutputCycle();
103 SaveDirName = col->getSaveDirName();
104 RestartDirName = col->getRestartDirName();
105 restart_status = col->getRestart_status();
106 ns = col->getNs(); // get the number of particle species involved in simulation
107 first_cycle = col->getLast_cycle() + 1; // get the last cycle from the restart
108 // initialize the virtual cartesian topology
109 vct = new VCtopology3D(*col);
110 // Check if we can map the processes into a matrix ordering defined in Collective.cpp
111 if (nprocs != vct->getNprocs() and myrank==0)
112 eprintf("Error: %i processes cant be mapped into %ix%ix%i matrix: Change XLEN,YLEN, ZLEN in method VCtopology3D.init()", nprocs, vct->getXLEN(), vct->getYLEN(), vct->getZLEN())
113
114 // We create a new communicator with a 3D virtual Cartesian topology
115 vct->setup_vctopology(MPIdata::get_PicGlobalComm());
116 {
117 stringstream num_proc_ss;
118 num_proc_ss << vct->getCartesian_rank();
119 num_proc_str = num_proc_ss.str();
120 }
121 // initialize the central cell index
122
123 // Print the initial settings to stdout and a file
124 if (myrank == 0) {
125 MPIdata::instance().Print();
126 vct->Print();
127 col->Print();
128 col->save();
129 }
130 // Create the local grid
131 grid = new Grid3DCU(col, vct); // Create the local grid
132 EMf = new EMfields3D(col, grid, vct); // Create Electromagnetic Fields Object
133
134 if (col->getCase()=="GEMnoPert") EMf->initGEMnoPert();
135 else if (col->getCase()=="ForceFree") EMf->initForceFree();
136 else if (col->getCase()=="GEM") EMf->initGEM();
137 else if (col->getCase()=="GEMDoubleHarris") EMf->initGEMDoubleHarris();
138 else if (col->getCase()=="Dipole") EMf->initDipole();
139 else if (col->getCase()=="Dipole2D") EMf->initDipole2D();
140 else if (col->getCase()=="NullPoints") EMf->initNullPoints();
141 else if (col->getCase()=="TaylorGreen") EMf->initTaylorGreen();
142 else {
143 if (myrank==0) {
144 cout << " =========================================================== " << endl;
145 cout << " WARNING: The case '" << col->getCase() << "' was not recognized. " << endl;
146 cout << " Runing simulation with the default initialization. " << endl;
147 cout << " =========================================================== " << endl;
148 }
149 EMf->init();
150 }
151
152 // Allocation of particles
153 part = (Particles3D*) malloc(sizeof(Particles3D)*ns);
154 for (int i = 0; i < ns; i++)
155 {
156 new(&part[i]) Particles3D(i,col,vct,grid);
157 }
158
159 // Initial Condition for PARTICLES if you are not starting from RESTART
160 if (restart_status == 0) {
161 for (int i = 0; i < ns; i++)
162 {
163 if (col->getCase()=="ForceFree") part[i].force_free(EMf);
164 else if (col->getCase()=="NullPoints") part[i].maxwellianNullPoints(EMf);
165 else if (col->getCase()=="TaylorGreen") part[i].maxwellianNullPoints(EMf); // Flow is initiated from the current prescribed on the grid.
166 else if (col->getCase()=="GEMDoubleHarris") part[i].maxwellianDoubleHarris(EMf);
167 else if (col->getCase()=="Dipole" and col->getNewPclInit()) part[i].maxwellianDipole(EMf,col->getL_square(),col->getx_center(),col->gety_center(),col->getz_center());
168 else if (col->getCase()=="Dipole2D" and col->getNewPclInit()) part[i].maxwellianDipole(EMf,col->getL_square(),col->getx_center(),0.,col->getz_center());
169 else part[i].maxwellian(EMf);
170 part[i].reserve_remaining_particle_IDs();
171 }
172 }
173
174 //allocate test particles if any
175 nstestpart = col->getNsTestPart();
176 if(nstestpart>0 && restart_status==0){
177 testpart = (Particles3D*) malloc(sizeof(Particles3D)*nstestpart);
178 for (int i = 0; i < nstestpart; i++)
179 {
180 new(&testpart[i]) Particles3D(i+ns,col,vct,grid);//species id for test particles is increased by ns
181 testpart[i].pitch_angle_energy(EMf);
182 }
183 }
184
185 if ( Parameters::get_doWriteOutput()){
186 #ifndef NO_HDF5
187 if(col->getWriteMethod() == "shdf5" || col->getCallFinalize() || restart_cycle>0 ||
188 (col->getWriteMethod()=="pvtk" && !col->particle_output_is_off()) )
189 {
190 outputWrapperFPP = new OutputWrapperFPP;
191 fetch_outputWrapperFPP().init_output_files(col,vct,grid,EMf,part,ns,testpart,nstestpart);
192 }
193 #endif
194 if(!col->field_output_is_off()){
195 if(col->getWriteMethod()=="pvtk"){
196 if(!(col->getFieldOutputTag()).empty())
197 fieldwritebuffer = newArr4(float,(grid->getNZN()-3),grid->getNYN()-3,grid->getNXN()-3,3);
198 if(!(col->getMomentsOutputTag()).empty())
199 momentwritebuffer=newArr3(float,(grid->getNZN()-3), grid->getNYN()-3, grid->getNXN()-3);
200 }
201 else if(col->getWriteMethod()=="nbcvtk"){
202 momentreqcounter=0;
203 fieldreqcounter = 0;
204 if(!(col->getFieldOutputTag()).empty())
205 fieldwritebuffer = newArr4(float,(grid->getNZN()-3)*4,grid->getNYN()-3,grid->getNXN()-3,3);
206 if(!(col->getMomentsOutputTag()).empty())
207 momentwritebuffer=newArr3(float,(grid->getNZN()-3)*14, grid->getNYN()-3, grid->getNXN()-3);
208 }
209 }
210 }
211 rho = new double[ns];
212 Ke = new double[ns];
213 BulkEnergy = new double[ns];
214 momentum = new double[ns];
215 cq = SaveDirName + "/ConservedQuantities.txt";
216 if (myrank == 0) {
217 ofstream my_file(cq.c_str());
218 my_file.close();
219 }
220
221 Qdel = new double[ns];
222 Count = new double[ns];
223 Qrep = new double[ns];
224 Qexo = new double[ns];
225 my_clock = new Timing(myrank);
226
227 return 0;
228}

230void c_Solver::CalculateMoments() {
231
232 timeTasks_set_main_task(TimeTasks::MOMENTS);
233
234 pad_particle_capacities();
235
236 // vectorized assumes that particles are sorted by mesh cell
237 if(Parameters::get_VECTORIZE_MOMENTS())
238 {
239 switch(Parameters::get_MOMENTS_TYPE())
240 {
241 case Parameters::SoA:
242 // since particles are sorted,
243 // we can vectorize interpolation of particles to grid
244 convertParticlesToSoA();
245 sortParticles();
246 EMf->sumMoments_vectorized(part);
247 break;
248 case Parameters::AoS:
249 convertParticlesToAoS();
250 sortParticles();
251 EMf->sumMoments_vectorized_AoS(part);
252 break;
253 default:
254 unsupported_value_error(Parameters::get_MOMENTS_TYPE());
255 }
256 }
257 else
258 {
259 if(Parameters::get_SORTING_PARTICLES())
260 sortParticles();
261 switch(Parameters::get_MOMENTS_TYPE())
262 {
263 case Parameters::SoA:
264 EMf->setZeroPrimaryMoments();
265 convertParticlesToSoA();
266 EMf->sumMoments(part);
267 break;
268 case Parameters::AoS:
269 EMf->setZeroPrimaryMoments();
270 convertParticlesToAoS();
271 EMf->sumMoments_AoS(part);
272 break;
273 case Parameters::AoSintr:
274 EMf->setZeroPrimaryMoments();
275 convertParticlesToAoS();
276 EMf->sumMoments_AoS_intr(part);
277 break;
278 default:
279 unsupported_value_error(Parameters::get_MOMENTS_TYPE());
280 }
281 }
282
283 EMf->setZeroDerivedMoments();
284 // sum all over the species
285 EMf->sumOverSpecies();
286 // Fill with constant charge the planet
287 if (col->getCase()=="Dipole") {
288 EMf->ConstantChargePlanet(col->getL_square(),col->getx_center(),col->gety_center(),col->getz_center());
289 if (col->getVerbose() and myrank==0) printf("**** Apply Constant Charge Inside Planet **** \n");
290 }else if(col->getCase()=="Dipole2D") {
291 EMf->ConstantChargePlanet2DPlaneXZ(col->getL_square(),col->getx_center(),col->getz_center());
292 if (col->getVerbose() and myrank==0) printf("**** Apply Constant Charge Inside Planet 2D **** \n");
293 }
294 // calculate densities on centers from nodes
295 EMf->interpDensitiesN2C();
296 // calculate the hat quantities for the implicit method
297 EMf->calculateHatFunctions();
298}

316/* -------------- */
317bool c_Solver::ParticlesMover(int cycle)
318{
319 // move all species of particles
320 {
321 timeTasks_set_main_task(TimeTasks::PARTICLES);
322 // Should change this to add background field
323 EMf->set_fieldForPcls();
324
325 pad_particle_capacities();
326
327 // Varibles for Exosphere injection
328 const double R = col->getL_square();
329 const double Nexo_H = 1e4; // density of exosphere neutrals at the surface (in nsw units)
330 const double fexo_H = 1e-9; // ioniz. frequency in units of wpi
331 const double hexo_H = 0.5*R; // scale length of exosphere
332 const double Nexo_Na = 1e3;
333 const double fexo_Na = 1e-7;
334 const double hexo_Na = 0.025*R;
335 const double w_fact = 8e3; // factor weight_exo / weight_sw
336
337 for (int i = 0; i < ns; i++) // move each species
338 {
339 // #pragma omp task inout(part[i]) in(grid) target_device(booster)
340 // should merely pass EMf->get_fieldForPcls() rather than EMf.
341 // use the Predictor Corrector scheme to move particles
342 switch(Parameters::get_MOVER_TYPE())
343 {
344 case Parameters::SoA:
345 part[i].mover_PC(EMf);
346 break;
347 case Parameters::AoS:
348 part[i].mover_PC_AoS(EMf);
349 break;
350 case Parameters::AoS_Relativistic:
351 part[i].mover_PC_AoS_Relativistic(EMf);
352 break;
353 case Parameters::AoSintr:
354 part[i].mover_PC_AoS_vec_intr(EMf);
355 break;
356 case Parameters::AoSvec:
357 part[i].mover_PC_AoS_vec(EMf);
358 break;
359 default:
360 unsupported_value_error(Parameters::get_MOVER_TYPE());
361 }
362
363 // Injection particles from ionized exosphere ./Job
364 // inject hydrogen
365 if ( (i==2 or i==3) and col->getAddExosphere()){
366 Qexo[i] = part[i].AddIonizedExosphere(R,col->getx_center(),col->gety_center(),col->getz_center(),Nexo_H,fexo_H,hexo_H,w_fact);
367 }
368 // inject sodium
369 if ( (i==4 or i==5) and col->getAddExosphere()){
370 Qexo[i] = part[i].AddIonizedExosphere(R,col->getx_center(),col->gety_center(),col->getz_center(),Nexo_Na,fexo_Na,hexo_Na,w_fact);
371 }
372
373 // External boundary conditions particles ./Job
374 Qrep[i] = part[i].repopulate_particles(EMf);
375 }
376
377 // Internal boundary conditions particles. ./Job
378 // case with re-inejction of pcls to keep net charge zero ./Job
379 double Qrm, Count_plus=0., Count_mins=0.;
380 if (col->getNonTrivialBCPlanet()) {
381 for (int i=0; i < ns; i++){
382 if (col->getCase()=="Dipole") Count[i] = part[i].rotateAndCountParticlesInsideSphere(cycle,R,col->getx_center(),col->gety_center(),col->getz_center());
383 else if (col->getCase()=="Dipole2D") Count[i] = part[i].rotateAndCountParticlesInsideSphere2DPlaneXZ(cycle,R,col->getx_center(),col->getz_center());
384 if (Count[i]>0) Count_plus += Count[i];
385 if (Count[i]<0) Count_mins += Count[i];
386 //if (col->getVerbose() and Count[i]!=0) dprintf("RotateAndCount->For proc %d the Q%i counted is = %f",myrank,i,Count[i]);
387 }
388 Qrm = std::min(Count_plus,-Count_mins);
389 }
390 // trivial case with only pcls removal ./Job
391 else {
392 Qrm = (double) INT_MAX;
393 }
394 for (int i=0; i < ns; i++) {
395 if (col->getCase()=="Dipole") Qdel[i] = part[i].deleteParticlesInsideSphere(cycle,Qrm,R,col->getx_center(),col->gety_center(),col->getz_center());
396 else if (col->getCase()=="Dipole2D") Qdel[i] = part[i].deleteParticlesInsideSphere2DPlaneXZ(cycle,Qrm,R,col->getx_center(),col->getz_center());
397 //if (col->getVerbose() and Qdel[i]!=0) dprintf("Delete->For proc %d the Q%i removed is = %f",myrank,i,Qdel[i]);
398 }
399
400 /* ---------------------------------- */
401 /* communicate pcls between procs */
402 /* --------------------------------- */
403 for (int i = 0; i < ns; i++){
404 part[i].separate_and_send_particles();
405 //part[i].communicate_particles();
406 part[i].recommunicate_particles_until_done(1);
407 }
408
409
410 /* --------------------------------------- */
411 /* Test Particles mover */
412 /* --------------------------------------- */
413 for (int i = 0; i < nstestpart; i++) // move each species
414 {
415 switch(Parameters::get_MOVER_TYPE())
416 {
417 case Parameters::SoA:
418 testpart[i].mover_PC(EMf);
419 break;
420 case Parameters::AoS:
421 testpart[i].mover_PC_AoS(EMf);
422 break;
423 case Parameters::AoS_Relativistic:
424 testpart[i].mover_PC_AoS_Relativistic(EMf);
425 break;
426 case Parameters::AoSintr:
427 testpart[i].mover_PC_AoS_vec_intr(EMf);
428 break;
429 case Parameters::AoSvec:
430 testpart[i].mover_PC_AoS_vec(EMf);
431 break;
432 default:
433 unsupported_value_error(Parameters::get_MOVER_TYPE());
434 }
435
436 testpart[i].openbc_delete_testparticles();
437 testpart[i].separate_and_send_particles();
438 testpart[i].recommunicate_particles_until_done(1);
439 }
440
441 return (false);
442 }
443}

446void c_Solver::WriteOutput(int cycle) {
447
448 WriteConserved(cycle);
449 WriteRestart(cycle);
450
451 if(!Parameters::get_doWriteOutput()) return;
452
453
454 if (col->getWriteMethod() == "nbcvtk"){//Non-blocking collective MPI-IO
455
456 if(!col->field_output_is_off() && (cycle%(col->getFieldOutputCycle()) == 0 || cycle == first_cycle) ){
457 if(!(col->getFieldOutputTag()).empty()){
458
459 if(fieldreqcounter>0){
460
461 //MPI_Waitall(fieldreqcounter,&fieldreqArr[0],&fieldstsArr[0]);
462 for(int si=0;si< fieldreqcounter;si++){
463 int error_code = MPI_File_write_all_end(fieldfhArr[si],&fieldwritebuffer[si][0][0][0],&fieldstsArr[si]);//fieldstsArr[si].MPI_ERROR;
464 if (error_code != MPI_SUCCESS) {
465 char error_string[100];
466 int length_of_error_string, error_class;
467 MPI_Error_class(error_code, &error_class);
468 MPI_Error_string(error_class, error_string, &length_of_error_string);
469 dprintf("MPI_Waitall error at field output cycle %d %d %s\n",cycle, si, error_string);
470 }else{
471 MPI_File_close(&(fieldfhArr[si]));
472 }
473 }
474 }
475 fieldreqcounter = WriteFieldsVTKNonblk(grid, EMf, col, vct,cycle,fieldwritebuffer,fieldreqArr,fieldfhArr);
476 }
477
478 if(!(col->getMomentsOutputTag()).empty()){
479
480 if(momentreqcounter>0){
481 //MPI_Waitall(momentreqcounter,&momentreqArr[0],&momentstsArr[0]);
482 for(int si=0;si< momentreqcounter;si++){
483 int error_code = MPI_File_write_all_end(momentfhArr[si],&momentwritebuffer[si][0][0],&momentstsArr[si]);//momentstsArr[si].MPI_ERROR;
484 if (error_code != MPI_SUCCESS) {
485 char error_string[100];
486 int length_of_error_string, error_class;
487 MPI_Error_class(error_code, &error_class);
488 MPI_Error_string(error_class, error_string, &length_of_error_string);
489 dprintf("MPI_Waitall error at moments output cycle %d %d %s\n",cycle, si, error_string);
490 }else{
491 MPI_File_close(&(momentfhArr[si]));
492 }
493 }
494 }
495 momentreqcounter = WriteMomentsVTKNonblk(grid, EMf, col, vct,cycle,momentwritebuffer,momentreqArr,momentfhArr);
496 }
497 }
498
499 //Particle information is still in hdf5
500 WriteParticles(cycle);
501 //Test Particle information is still in hdf5
502 WriteTestParticles(cycle);
503
504 }else if (col->getWriteMethod() == "pvtk"){//Blocking collective MPI-IO
505 if(!col->field_output_is_off() && (cycle%(col->getFieldOutputCycle()) == 0 || cycle == first_cycle) ){
506 if(!(col->getFieldOutputTag()).empty()){
507 //WriteFieldsVTK(grid, EMf, col, vct, col->getFieldOutputTag() ,cycle);//B + E + Je + Ji + rho
508 WriteFieldsVTK(grid, EMf, col, vct, col->getFieldOutputTag() ,cycle, fieldwritebuffer);//B + E + Je + Ji + rho
509 }
510 if(!(col->getMomentsOutputTag()).empty()){
511 WriteMomentsVTK(grid, EMf, col, vct, col->getMomentsOutputTag() ,cycle, momentwritebuffer);
512 }
513 }
514
515 //Particle information is still in hdf5
516 WriteParticles(cycle);
517 //Test Particle information is still in hdf5
518 WriteTestParticles(cycle);
519
520 }else{
521
522 #ifdef NO_HDF5
523 eprintf("The selected output option must be compiled with HDF5");
524
525 #else
526 if (col->getWriteMethod() == "H5hut"){
527
528 if (!col->field_output_is_off() && cycle%(col->getFieldOutputCycle())==0)
529 WriteFieldsH5hut(ns, grid, EMf, col, vct, cycle);
530 if (!col->particle_output_is_off() && cycle%(col->getParticlesOutputCycle())==0)
531 WritePartclH5hut(ns, grid, part, col, vct, cycle);
532
533 }else if (col->getWriteMethod() == "phdf5"){
534
535 if (!col->field_output_is_off() && cycle%(col->getFieldOutputCycle())==0)
536 WriteOutputParallel(grid, EMf, part, col, vct, cycle);
537
538 if (!col->particle_output_is_off() && cycle%(col->getParticlesOutputCycle())==0)
539 {
540 if(MPIdata::get_rank()==0)
541 warning_printf("WriteParticlesParallel() is not yet implemented.");
542 }
543
544 }else if (col->getWriteMethod() == "shdf5"){
545
546 WriteFields(cycle);
547
548 WriteParticles(cycle);
549
550 WriteTestParticles(cycle);
551
552 }else{
553 warning_printf(
554 "Invalid output option. Options are: H5hut, phdf5, shdf5, pvtk");
555 invalid_value_error(col->getWriteMethod().c_str());
556 }
557 #endif
558 }
559}

571// write the conserved quantities
572void c_Solver::WriteConserved(int cycle) {
573 if(col->getDiagnosticsOutputCycle() > 0 && cycle % col->getDiagnosticsOutputCycle() == 0) {
574 Eenergy = EMf->getEenergy();
575 Benergy = EMf->getBenergy();
576 TOTenergy = 0.0;
577 TOTmomentum = 0.0;
578 for (int is = 0; is < ns; is++) {
579 rho[is]= part[is].getRho();
580 Ke[is] = part[is].getKe();
581 BulkEnergy[is] = EMf->getBulkEnergy(is);
582 TOTenergy += Ke[is];
583 momentum[is] = part[is].getP();
584 TOTmomentum += momentum[is];
585 }
586 if (myrank == (nprocs-1)) {
587 ofstream my_file(cq.c_str(), fstream::app);
588 if(cycle==0){
589 my_file << "cycle" << "\t" << "Total_Energy" << "\t" << "Momentum" << "\t" << "Eenergy" << "\t" << "Benergy" << "\t" << "Kenergy" ;
590 for (int is = 0; is < ns; is++) my_file << "Kenergy("<<is<<")" << "\t";
591 for (int is = 0; is < ns; is++) my_file << "BulkEnergy("<<is<<")" << "\t";
592 for (int is = 0; is < ns; is++) my_file << "Rho("<<is<<")" << "\t";
593 my_file << endl;
594 }
595 my_file << cycle << "\t" << (Eenergy + Benergy + TOTenergy) << "\t" << TOTmomentum << "\t" << Eenergy << "\t" << Benergy << "\t" << TOTenergy;
596 for (int is = 0; is < ns; is++) my_file << "\t" << Ke[is];
597 for (int is = 0; is < ns; is++) my_file << "\t" << BulkEnergy[is];
598 for (int is = 0; is < ns; is++) my_file << "\t" << rho[is];
599 my_file << endl;
600 my_file.close();
