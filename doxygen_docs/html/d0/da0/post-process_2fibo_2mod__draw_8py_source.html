20 def draw_canvas(self,
21 tar_labs,
22 tar_dims,
23 tar_char=12):
24 """
25 Prepares canvas for drawing
26
27 Parameters :
28 - tar_labs [list of list of str] matrix of labels
29 - tar_dims [6*[float] OR 'line' OR 'cont_h' OR 'cont_v'] dims of figs and spaces within
30 - tar_char [float] dimension of the ticks and labels
31 Returns :
32 - tar_plot [blank figure]
33
34 """
35
36 y_fig_num, x_fig_num = np.shape(tar_labs)
37
38 if tar_dims is 'line' : tar_dims = [7.2,3.0,0.8,0.6,0.6,0.6]
39 if tar_dims is 'cont_h' : tar_dims = [5.2,5.2,0.8,0.6,0.6,0.6]
40 if tar_dims is 'cont_v' : tar_dims = [6.0,6.0,0.6,0.6,0.6,0.6]
41
42 dim_x = tar_dims[0]*x_fig_num + tar_dims[2]*(x_fig_num-1) + 2*tar_dims[4]
43 dim_y = tar_dims[1]*y_fig_num + tar_dims[3]*(y_fig_num-1) + 2*tar_dims[5]
44
45 multi_fig = plt.figure(figsize=(dim_x,dim_y))
46 multi_fig.patch.set_facecolor('white')
47
48 for ii in range(y_fig_num):
49 for jj in range(x_fig_num):
50 if tar_labs[ii][jj] is not None :
51
52 pos_x = (tar_dims[4] + (tar_dims[0] + tar_dims[2]) * jj)/dim_x
53 pos_y = (tar_dims[5] + (tar_dims[1] + tar_dims[3]) * (y_fig_num-1-ii) )/dim_y
54
55 multi_fig.add_axes([pos_x,pos_y,tar_dims[0]/dim_x,tar_dims[1]/dim_y])
56 mpl.pyplot.ylabel(tar_labs[ii][jj])
57
58 tar_plot = multi_fig.get_axes()
59
60 for ax in tar_plot :
61 ax.xaxis.set_tick_params(labelsize=tar_char)
62 ax.yaxis.set_tick_params(labelsize=tar_char)
63
64 return tar_plot
