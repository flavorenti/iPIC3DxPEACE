52void MPIdata::init(int *argc, char ***argv) {
53 assert(!MPIdata_is_initialized);
54
55 #ifdef NO_MPI
56 rank = 0;
57 nprocs = 1;
58 #else // NO_MPI
59 /* Initialize the MPI API */
60 MPI_Init(argc, argv);
61
62 MPI_Comm_dup(MPI_COMM_WORLD, &PIC_COMM);
63 MPI_Comm_rank(PIC_COMM, &rank);
64 MPI_Comm_size(PIC_COMM, &nprocs);
65
66 #endif // NO_MPI
67
68 MPIdata_is_initialized = true;
69}
